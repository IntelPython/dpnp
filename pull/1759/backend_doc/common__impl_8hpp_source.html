39template <typename EngineBuilderT,
40 typename DistributorBuilderT,
41 unsigned int items_per_wi = 4,
42 bool enable_sg_load = true>
43struct RngContigFunctor
44{
45private:
46 using DataT = typename DistributorBuilderT::result_type;
47
48 EngineBuilderT engine_;
49 DistributorBuilderT distr_;
50 DataT *const res_ = nullptr;
51 const std::size_t nelems_;
52
53public:
54 RngContigFunctor(EngineBuilderT &engine,
55 DistributorBuilderT &distr,
56 DataT *res,
57 const std::size_t n_elems)
58 : engine_(engine), distr_(distr), res_(res), nelems_(n_elems)
59 {
60 }
61
62 void operator()(sycl::nd_item<1> nd_it) const
63 {
64 auto sg = nd_it.get_sub_group();
65 const std::uint8_t sg_size = sg.get_local_range()[0];
66 const std::uint8_t max_sg_size = sg.get_max_local_range()[0];
67
68 using EngineT = typename EngineBuilderT::EngineType;
69 using DistrT = typename DistributorBuilderT::distr_type;
70
71 constexpr std::size_t vec_sz = EngineT::vec_size;
72 constexpr std::size_t vi_per_wi = vec_sz * items_per_wi;
73
74 EngineT engine = engine_(nd_it.get_global_id() * vi_per_wi);
75 DistrT distr = distr_();
76
77 if constexpr (enable_sg_load) {
78 const std::size_t base =
79 vi_per_wi * (nd_it.get_group(0) * nd_it.get_local_range(0) +
80 sg.get_group_id()[0] * max_sg_size);
81
82 if ((sg_size == max_sg_size) &&
83 (base + vi_per_wi * sg_size < nelems_)) {
84#pragma unroll
85 for (std::uint16_t it = 0; it < vi_per_wi; it += vec_sz) {
86 std::size_t offset =
87 base + static_cast<std::size_t>(it) *
88 static_cast<std::size_t>(sg_size);
89 auto out_multi_ptr = sycl::address_space_cast<
90 sycl::access::address_space::global_space,
91 sycl::access::decorated::yes>(&res_[offset]);
92
93 sycl::vec<DataT, vec_sz> rng_val_vec =
94 mkl_rng_dev::generate<DistrT, EngineT>(distr, engine);
95 sg.store<vec_sz>(out_multi_ptr, rng_val_vec);
96 }
97 }
98 else {
99 for (std::size_t offset = base + sg.get_local_id()[0];
100 offset < nelems_; offset += sg_size)
101 {
102 res_[offset] =
103 mkl_rng_dev::generate_single<DistrT, EngineT>(distr,
104 engine);
105 }
106 }
107 }
108 else {
109 std::size_t base = nd_it.get_global_linear_id();
110
111 base = (base / sg_size) * sg_size * vi_per_wi + (base % sg_size);
112 for (std::size_t offset = base;
113 offset < std::min(nelems_, base + sg_size * vi_per_wi);
114 offset += sg_size)
115 {
116 res_[offset] = mkl_rng_dev::generate_single<DistrT, EngineT>(
117 distr, engine);
118 }
119 }
120 }
121};
122} // namespace dpnp::backend::ext::rng::device::details
