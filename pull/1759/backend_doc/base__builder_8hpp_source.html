36template <typename EngineT, typename SeedT, typename OffsetT>
37class BaseBuilder
38{
39private:
40 static constexpr std::uint8_t max_n = EngineBase::max_vec_n;
41
42 std::uint8_t no_of_seeds;
43 std::uint8_t no_of_offsets;
44
45 std::array<SeedT, max_n> seeds{};
46 std::array<OffsetT, max_n> offsets{};
47
48public:
49 BaseBuilder(EngineBase *engine)
50 {
51 auto seed_values = engine->get_seeds();
52 no_of_seeds = seed_values.size();
53 if (no_of_seeds > max_n) {
54 throw std::runtime_error("");
55 }
56
57 // TODO: implement a caster
58 for (std::uint16_t i = 0; i < no_of_seeds; i++) {
59 seeds[i] = static_cast<SeedT>(seed_values[i]);
60 }
61
62 auto offset_values = engine->get_offsets();
63 no_of_offsets = offset_values.size();
64 if (no_of_offsets > max_n) {
65 throw std::runtime_error("");
66 }
67
68 // TODO: implement a caster
69 for (std::uint16_t i = 0; i < no_of_seeds; i++) {
70 offsets[i] = static_cast<OffsetT>(offset_values[i]);
71 }
72 }
73
74 inline auto operator()(void) const
75 {
76 switch (no_of_seeds) {
77 case 1:
78 {
79 if constexpr (std::is_same_v<EngineT,
80 mkl_rng_dev::mcg59<EngineT::vec_size>>)
81 {
82 // issue with mcg59<>() constructor which breaks compilation
83 return EngineT(seeds[0], offsets[0]);
84 }
85 else {
86 return EngineT({seeds[0]}, offsets[0]);
87 }
88 }
89 // TODO: implement full switch
90 default:
91 break;
92 }
93 return EngineT();
94 }
95
96 inline auto operator()(const OffsetT offset) const
97 {
98 switch (no_of_seeds) {
99 case 1:
100 {
101 if constexpr (std::is_same_v<EngineT,
102 mkl_rng_dev::mcg59<EngineT::vec_size>>)
103 {
104 // issue with mcg59<>() constructor which breaks compilation
105 return EngineT(seeds[0], offsets[0] + offset);
106 }
107 else {
108 return EngineT({seeds[0]}, {offsets[0] + offset});
109 }
110 }
111 // TODO: implement full switch
112 default:
113 break;
114 }
115 return EngineT();
116 }
