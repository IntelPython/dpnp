Interplay with the Data Parallel Control Library

Data Parallel Control Library provides API to manage specific SYCL* resources for SYCL-based Python packages.

An example below demonstrates how the Data Parallel Extension for NumPy* can be easily combined with the device management interface provided by dpctl package.

 1    import dpctl
 2    import dpnp
 3
 4    d = dpctl.select_cpu_device()
 5    x = dpnp.array([1, 2, 3], device=d)
 6    s = dpnp.sum(x)
 7
 8    y = dpnp.linspace(0, dpnp.pi, num=10**6, device="gpu")
 9    f = 1 + y * dpnp.sin(y)
10
11    # locate argument where function attains global maximum
12    max_arg = x[dpnp.argmax(f)]
13    max_val = dpnp.max(f)

For more information please refer to Data Parallel Control Library documentation.

10def run(executor, size, test_type, repetition):
11    x = executor.reshape(
12        executor.arange(size * size, dtype=test_type), (size, size)
13    )
14
15    times = []
16    for _ in range(repetition):
17        start_time = time.perf_counter()
18        result = executor.sum(x)
19        end_time = time.perf_counter()
20        times.append(end_time - start_time)
21
22    return numpy.median(times), result
23

25def get_dtypes():
26    _dtypes_list = [numpy.int32, numpy.int64, numpy.float32]
27    device = dpctl.select_default_device()
28    if device.has_aspect_fp64:
29        _dtypes_list.append(numpy.float64)
30    return _dtypes_list
31

33def example():
34    test_repetition = 5
35    for test_type in get_dtypes():
36        type_name = numpy.dtype(test_type).name
37        print(
38            f"...Test data type is {type_name}, each test repetitions {test_repetition}"
39        )
40
41        for size in [64, 128, 256, 512, 1024, 2048, 4096]:
42            time_numpy, result_numpy = run(
43                numpy, size, test_type, test_repetition
44            )
45            time_dpnp, result_dpnp = run(dpnp, size, test_type, test_repetition)
46
47            if result_dpnp == result_numpy:
48                verification = True
49            else:
50                verification = f"({result_dpnp} != {result_numpy})"
51
52            msg = f"type:{type_name}:N:{size:4}:NumPy:{time_numpy:.3e}:DPNP:{time_dpnp:.3e}"
53            msg += f":ratio:{time_numpy/time_dpnp:6.2f}:verification:{verification}"
