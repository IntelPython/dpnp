47bool check_limitations(const dpctl::tensor::usm_ndarray &in,
48 const dpctl::tensor::usm_ndarray &out,
49 bool throw_on_fail = false)
50{
51 if (not in.is_c_contiguous() or not out.is_c_contiguous()) {
52 if (throw_on_fail)
53 throw py::value_error(
54 "Input and output arrays must be c-contiguos");
55
56 return false;
57 }
58
59 auto device = in.get_queue().get_device();
60 auto local_mem_size = device.get_info<sycl::info::device::local_mem_size>();
61 size_t out_full_size = out.get_size() * out.get_elemsize();
62 if (out_full_size > local_mem_size) {
63 if (throw_on_fail)
64 throw py::value_error("Resulting array exceeds local memory size" +
65 std::to_string(local_mem_size));
66
67 return false;
68 }
69
70 if (out.get_elemsize() == 64 and not device.has(sycl::aspect::atomic64)) {
71 if (throw_on_fail)
72 throw py::value_error("64-bit atomics are not supported");
73
74 return false;
75 }
76
77 return true;
78}

84template <typename InT, typename OutT, bool Mean>
85struct sum_mean
86{
87 static sycl::event call(sycl::queue exec_q,
88 const InT *in,
89 OutT *out,
90 const size_t height,
91 const size_t width,
92 const std::vector<sycl::event> &depends)
93 {
94 int local_size = std::min(
95 1024,
96 int(exec_q.get_device()
97 .get_info<sycl::info::device::max_work_group_size>()));
98
99 int WorkPI = 32; // empirically found number
100 auto _total_size = height * width;
101 auto global_size = Align(DivUp(_total_size, WorkPI), local_size);
102
103 auto fill_e = exec_q.fill(out, OutT(0), width, depends);
104
105 return exec_q.submit([&](sycl::handler &cgh) {
106 cgh.depends_on(fill_e);
107 auto local_result =
108 sycl::local_accessor<OutT>(sycl::range<1>(width), cgh);
109 cgh.parallel_for<sum_mean<InT, OutT, Mean>>(
110 sycl::nd_range<1>(global_size, local_size),
111 [=](sycl::nd_item<1> item) {
112 auto grid = item.get_group_linear_id();
113 auto group = item.get_group();
114 auto llid = item.get_local_linear_id();
115
116 for (size_t i = llid; i < width; i += local_size) {
117 local_result[i] = 0;
118 }
119
120 sycl::group_barrier(group);
121
122 for (int i = 0; i < WorkPI; ++i) {
123 auto id =
124 grid * WorkPI * local_size + i * local_size + llid;
125 if (id < _total_size) {
126 sycl::atomic_ref<OutT, sycl::memory_order::relaxed,
127 sycl::memory_scope::work_group>
128 aresult(local_result[id % width]);
129 aresult += in[id];
130 }
131 }
132
133 sycl::group_barrier(group);
134
135 for (size_t i = llid; i < width; i += local_size) {
136 sycl::atomic_ref<OutT, sycl::memory_order::relaxed,
137 sycl::memory_scope::device>
138 autput(out[i]);
139
140 auto r = local_result[i];
141 autput += Mean ? r / height : r;
142 }
143 });
144 });
145 }

147 static sycl::event call(const dpctl::tensor::usm_ndarray &in,
148 dpctl::tensor::usm_ndarray &out,
149 const std::vector<sycl::event> &depends)
150 {
151 validate_params(in, out);
152 return call(in.get_queue(), in.get_data<InT>(), out.get_data<OutT>(),
153 in.get_shape(0), in.get_shape(1), depends);
154 }
155
156 static void validate_params(const dpctl::tensor::usm_ndarray &in,
157 dpctl::tensor::usm_ndarray &out)
158 {
159 auto exec_q = in.get_queue();
160
161 type_utils::validate_type_for_device<InT>(exec_q);
162 type_utils::validate_type_for_device<OutT>(exec_q);
163
164 // check compatibility of execution queue and allocation queue
165 if (!dpctl::utils::queues_are_compatible(exec_q, {in, out})) {
166 throw py::value_error(
167 "Execution queue is not compatible with allocation queues");
168 }
169
170 if (in.get_shape(1) != out.get_shape(0))
171 throw py::value_error(
172 "Input array axis 1 size must match output array size");
173
174 auto const &overlap = dpctl::tensor::overlap::MemoryOverlap();
175 if (overlap(in, out))
176 throw py::value_error("Input and output array are overlapped");
177
178 check_limitations(in, out, true);
179 }
180};
