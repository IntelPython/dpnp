41template <typename T, typename RealT>
42static sycl::event syevd_impl(sycl::queue &exec_q,
43 const oneapi::mkl::job jobz,
44 const oneapi::mkl::uplo upper_lower,
45 const std::int64_t n,
46 char *in_a,
47 char *out_w,
48 std::vector<sycl::event> &host_task_events,
49 const std::vector<sycl::event> &depends)
50{
51 type_utils::validate_type_for_device<T>(exec_q);
52 type_utils::validate_type_for_device<RealT>(exec_q);
53
54 T *a = reinterpret_cast<T *>(in_a);
55 RealT *w = reinterpret_cast<T *>(out_w);
56
57 const std::int64_t lda = std::max<size_t>(1UL, n);
58 const std::int64_t scratchpad_size =
59 mkl_lapack::syevd_scratchpad_size<T>(exec_q, jobz, upper_lower, n, lda);
60 T *scratchpad = nullptr;
61
62 std::stringstream error_msg;
63 std::int64_t info = 0;
64
65 sycl::event syevd_event;
66 try {
67 scratchpad = sycl::malloc_device<T>(scratchpad_size, exec_q);
68
69 syevd_event = mkl_lapack::syevd(
70 exec_q,
71 jobz, // 'jobz == job::vec' means eigenvalues and eigenvectors are
72 // computed.
73 upper_lower, // 'upper_lower == job::upper' means the upper
74 // triangular part of A, or the lower triangular
75 // otherwise
76 n, // The order of the matrix A (0 <= n)
77 a, // Pointer to A, size (lda, *), where the 2nd dimension, must be
78 // at least max(1, n) If 'jobz == job::vec', then on exit it will
79 // contain the eigenvectors of A
80 lda, // The leading dimension of a, must be at least max(1, n)
81 w, // Pointer to array of size at least n, it will contain the
82 // eigenvalues of A in ascending order
83 scratchpad, // Pointer to scratchpad memory to be used by MKL
84 // routine for storing intermediate results
85 scratchpad_size, depends);
86 } catch (mkl_lapack::exception const &e) {
87 error_msg
88 << "Unexpected MKL exception caught during syevd() call:\nreason: "
89 << e.what() << "\ninfo: " << e.info();
90 info = e.info();
91 } catch (sycl::exception const &e) {
92 error_msg << "Unexpected SYCL exception caught during syevd() call:\n"
93 << e.what();
94 info = -1;
95 }
96
97 if (info != 0) // an unexpected error occurs
98 {
99 if (scratchpad != nullptr) {
100 sycl::free(scratchpad, exec_q);
101 }
102 throw std::runtime_error(error_msg.str());
103 }
104
105 sycl::event clean_up_event = exec_q.submit([&](sycl::handler &cgh) {
106 cgh.depends_on(syevd_event);
107 auto ctx = exec_q.get_context();
108 cgh.host_task([ctx, scratchpad]() { sycl::free(scratchpad, ctx); });
109 });
110
111 host_task_events.push_back(clean_up_event);
112 return syevd_event;
113}
