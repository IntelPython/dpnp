74template <typename Tp>
75inline Tp cyl_bessel_ij_0_series(Tp x, Tp sgn, unsigned int max_iter)
76{
77 if (x == Tp(0)) {
78 return Tp(1);
79 }
80
81 const Tp x2 = x / Tp(2);
82 Tp fact = -sycl::lgamma(Tp(1));
83 fact = std::exp(fact);
84
85 const Tp xx4 = sgn * x2 * x2;
86 Tp Jn = Tp(1);
87 Tp term = Tp(1);
88
89 for (unsigned int i = 1; i < max_iter; ++i) {
90 term *= xx4 / (Tp(i) * Tp(i));
91 Jn += term;
92 if (sycl::fabs(term / Jn) < std::numeric_limits<Tp>::epsilon()) {
93 break;
94 }
95 }
96 return fact * Jn;
97}

145template <typename Tp>
146inline void gamma_temme(Tp mu, Tp &gam1, Tp &gam2, Tp &gampl, Tp &gammi)
147{
148 gampl = Tp(1) / sycl::tgamma(Tp(1) + mu);
149 gammi = Tp(1) / sycl::tgamma(Tp(1) - mu);
150
151 if (sycl::fabs(mu) < std::numeric_limits<Tp>::epsilon())
152 // constant Euler's constant @f$ \gamma_E @f$.
153 gam1 = -static_cast<Tp>(0.5772156649015328606065120900824024L);
154 else
155 gam1 = (gammi - gampl) / (Tp(2) * mu);
156
157 gam2 = (gammi + gampl) / (Tp(2));
158}

175template <typename Tp>
176inline void bessel_ik_0(Tp x, Tp &Inu, Tp &Knu, Tp &Ipnu, Tp &Kpnu)
177{
178 if (x == Tp(0)) {
179 bessel_ik_0_x_0(Inu, Knu, Ipnu, Kpnu);
180 return;
181 }
182
183 constexpr Tp eps = std::numeric_limits<Tp>::epsilon();
184 constexpr Tp fp_min = Tp(10) * eps;
185 constexpr int max_iter = 15000;
186 constexpr Tp x_min = Tp(2);
187
188 constexpr int nl = int(0.5L);
189
190 const Tp mu = -nl;
191 const Tp mu2 = mu * mu;
192 const Tp xi = Tp(1) / x;
193 const Tp xi2 = Tp(2) * xi;
194 Tp h = fp_min;
195
196 Tp b = Tp(0);
197 Tp d = Tp(0);
198 Tp c = h;
199 int i;
200 for (i = 1; i <= max_iter; ++i) {
201 b += xi2;
202 d = Tp(1) / (b + d);
203 c = b + Tp(1) / c;
204
205 const Tp del = c * d;
206 h *= del;
207 if (sycl::fabs(del - Tp(1)) < eps) {
208 break;
209 }
210 }
211 if (i > max_iter) {
212 // argument `x` is too large
213 bessel_ik_0_x_0(Inu, Knu, Ipnu, Kpnu);
214 return;
215 }
216
217 Tp Inul = fp_min;
218 Tp Ipnul = h * Inul;
219 Tp Inul1 = Inul;
220 Tp Ipnu1 = Ipnul;
221 Tp fact = Tp(0);
222 for (int l = nl; l >= 1; --l) {
223 const Tp Inutemp = fact * Inul + Ipnul;
224 fact -= xi;
225 Ipnul = fact * Inutemp + Inul;
226 Inul = Inutemp;
227 }
228
229 constexpr Tp pi = static_cast<Tp>(3.1415926535897932384626433832795029L);
230 Tp f = Ipnul / Inul;
231 Tp Kmu, Knu1;
232 if (x < x_min) {
233 const Tp x2 = x / Tp(2);
234 const Tp pimu = pi * mu;
235 const Tp fact =
236 (sycl::fabs(pimu) < eps ? Tp(1) : pimu / sycl::sin(pimu));
237
238 Tp d = -sycl::log(x2);
239 Tp e = mu * d;
240 const Tp fact2 = (sycl::fabs(e) < eps ? Tp(1) : sycl::sinh(e) / e);
241
242 Tp gam1, gam2, gampl, gammi;
243 gamma_temme(mu, gam1, gam2, gampl, gammi);
244
245 Tp ff = fact * (gam1 * sycl::cosh(e) + gam2 * fact2 * d);
246 Tp sum = ff;
247 e = sycl::exp(e);
248
249 Tp p = e / (Tp(2) * gampl);
250 Tp q = Tp(1) / (Tp(2) * e * gammi);
251 Tp c = Tp(1);
252 d = x2 * x2;
253 Tp sum1 = p;
254 int i;
255 for (i = 1; i <= max_iter; ++i) {
256 ff = (i * ff + p + q) / (i * i - mu2);
257 c *= d / i;
258 p /= i - mu;
259 q /= i + mu;
260 const Tp del = c * ff;
261 sum += del;
262 const Tp __del1 = c * (p - i * ff);
263 sum1 += __del1;
264 if (sycl::fabs(del) < eps * sycl::fabs(sum)) {
265 break;
266 }
267 }
268 if (i > max_iter) {
269 // Bessel k series failed to converge
270 bessel_ik_0_x_0(Inu, Knu, Ipnu, Kpnu);
271 return;
272 }
273 Kmu = sum;
274 Knu1 = sum1 * xi2;
275 }
276 else {
277 Tp b = Tp(2) * (Tp(1) + x);
278 Tp d = Tp(1) / b;
279 Tp delh = d;
280 Tp h = delh;
281 Tp q1 = Tp(0);
282 Tp q2 = Tp(1);
283 Tp a1 = Tp(0.25L) - mu2;
284 Tp q = c = a1;
285 Tp a = -a1;
286 Tp s = Tp(1) + q * delh;
287 int i;
288 for (i = 2; i <= max_iter; ++i) {
289 a -= 2 * (i - 1);
290 c = -a * c / i;
291 const Tp qnew = (q1 - b * q2) / a;
292 q1 = q2;
293 q2 = qnew;
294 q += c * qnew;
295 b += Tp(2);
296 d = Tp(1) / (b + a * d);
297 delh = (b * d - Tp(1)) * delh;
298 h += delh;
299 const Tp dels = q * delh;
300 s += dels;
301 if (sycl::fabs(dels / s) < eps) {
302 break;
303 }
304 }
305 if (i > max_iter) {
306 // Steed's method failed
307 bessel_ik_0_x_0(Inu, Knu, Ipnu, Kpnu);
308 return;
309 }
310 h = a1 * h;
311 Kmu = std::sqrt(pi / (Tp(2) * x)) * std::exp(-x) / s;
312 Knu1 = Kmu * (mu + x + Tp(0.5L) - h) * xi;
313 }
314
315 Tp Kpmu = mu * xi * Kmu - Knu1;
316 Tp Inumu = xi / (f * Kmu - Kpmu);
317 Inu = Inumu * Inul1 / Inul;
318 Ipnu = Inumu * Ipnu1 / Inul;
319 for (i = 1; i <= nl; ++i) {
320 const Tp Knutemp = (mu + i) * xi2 * Knu1 + Kmu;
321 Kmu = Knu1;
322 Knu1 = Knutemp;
323 }
324 Knu = Kmu;
325 Kpnu = -Knu1;
326}

341template <typename Tp>
342inline Tp cyl_bessel_0(Tp x)
343{
344 if (sycl::isnan(x)) {
345 return std::numeric_limits<Tp>::quiet_NaN();
346 }
347
348 x = sycl::fabs(x);
349
350 if (x * x < Tp(10)) {
351 return cyl_bessel_ij_0_series<Tp>(x, +Tp(1), 200);
352 }
353 else {
354 Tp I_nu, K_nu, Ip_nu, Kp_nu;
355 bessel_ik_0<Tp>(x, I_nu, K_nu, Ip_nu, Kp_nu);
356 return I_nu;
357 }
