60template <mkl_dft::precision prec>
61class ComplexDescriptorWrapper
62{
63public:
64 ComplexDescriptorWrapper(std::int64_t n) : descr_(n), queue_ptr_{} {}
65 ComplexDescriptorWrapper(std::vector<std::int64_t> dimensions)
66 : descr_(dimensions), queue_ptr_{}
67 {
68 }
69 ~ComplexDescriptorWrapper() {}
70
71 void commit(sycl::queue &q)
72 {
73 mkl_dft::precision fft_prec = get_precision();
74 if (fft_prec == mkl_dft::precision::DOUBLE &&
75 !q.get_device().has(sycl::aspect::fp64))
76 {
77 throw py::value_error("Descriptor is double precision but the "
78 "device does not support double precision.");
79 }
80
81 descr_.commit(q);
82 queue_ptr_ = std::make_unique<sycl::queue>(q);
83 }
84
85 mkl_dft::descriptor<prec, mkl_dft::domain::COMPLEX> &get_descriptor()
86 {
87 return descr_;
88 }
89
90 const sycl::queue &get_queue() const
91 {
92 if (queue_ptr_) {
93 return *queue_ptr_;
94 }
95 else {
96 throw std::runtime_error(
97 "Attempt to get queue when it is not yet set");
98 }
99 }

254template <mkl_dft::precision prec>
255std::pair<sycl::event, sycl::event>
256 compute_fft(ComplexDescriptorWrapper<prec> &descr,
257 const dpctl::tensor::usm_ndarray &in,
258 const dpctl::tensor::usm_ndarray &out,
259 const bool is_forward,
260 const std::vector<sycl::event> &depends)
261{
262 // TODO: activate in MKL=2024.2
263 // bool committed = descr.is_committed();
264 // if (!committed) {
265 // throw py::value_error("Descriptor is not committed");
266 //}
267
268 const bool in_place = descr.get_in_place();
269 if (in_place) {
270 throw py::value_error(
271 "Descriptor is in-place while config_param::PLACEMENT is set to be "
272 "out-of-place.");
273 }
274
275 const int in_nd = in.get_ndim();
276 const int out_nd = out.get_ndim();
277 if ((in_nd != out_nd)) {
278 throw py::value_error(
279 "The input and output arrays must have the same dimension.");
280 }
281
282 auto const &overlap = dpctl::tensor::overlap::MemoryOverlap();
283 if (overlap(in, out)) {
284 throw py::value_error("The input and output arrays are overlapping "
285 "segments of memory");
286 }
287
288 sycl::queue exec_q = descr.get_queue();
289 if (!dpctl::utils::queues_are_compatible(exec_q,
290 {in.get_queue(), out.get_queue()}))
291 {
292 throw py::value_error(
293 "USM allocations are not compatible with the execution queue.");
294 }
295
296 py::ssize_t in_size = in.get_size();
297 py::ssize_t out_size = out.get_size();
298 if (in_size != out_size) {
299 throw py::value_error("The size of the input vector must be "
300 "equal to the size of the output vector.");
301 }
302
303 size_t src_nelems = in_size;
304 dpctl::tensor::validation::CheckWritable::throw_if_not_writable(out);
305 dpctl::tensor::validation::AmpleMemory::throw_if_not_ample(out, src_nelems);
306
307 using ScaleT = typename ScaleType<prec>::value_type;
308 std::complex<ScaleT> *in_ptr = in.get_data<std::complex<ScaleT>>();
309 std::complex<ScaleT> *out_ptr = out.get_data<std::complex<ScaleT>>();
310
311 sycl::event fft_event = {};
312 std::stringstream error_msg;
313 bool is_exception_caught = false;
314
315 try {
316 if (is_forward) {
317 fft_event = oneapi::mkl::dft::compute_forward(
318 descr.get_descriptor(), in_ptr, out_ptr, depends);
319 }
320 else {
321 fft_event = oneapi::mkl::dft::compute_backward(
322 descr.get_descriptor(), in_ptr, out_ptr, depends);
323 }
324 } catch (oneapi::mkl::exception const &e) {
325 error_msg
326 << "Unexpected MKL exception caught during FFT() call:\nreason: "
327 << e.what();
328 is_exception_caught = true;
329 } catch (sycl::exception const &e) {
330 error_msg << "Unexpected SYCL exception caught during FFT() call:\n"
331 << e.what();
332 is_exception_caught = true;
333 }
334 if (is_exception_caught) {
335 throw std::runtime_error(error_msg.str());
336 }
337
338 sycl::event args_ev =
339 dpctl::utils::keep_args_alive(exec_q, {in, out}, {fft_event});
340
341 return std::make_pair(fft_event, args_ev);
342}
