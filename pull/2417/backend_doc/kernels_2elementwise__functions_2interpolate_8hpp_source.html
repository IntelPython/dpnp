35template <typename TCoord, typename TValue>
36sycl::event interpolate_impl(sycl::queue &q,
37 const TCoord *x,
38 const std::int64_t *idx,
39 const TCoord *xp,
40 const TValue *fp,
41 const TValue *left,
42 const TValue *right,
43 TValue *out,
44 const std::size_t n,
45 const std::size_t xp_size,
46 const std::vector<sycl::event> &depends)
47{
48 return q.submit([&](sycl::handler &h) {
49 h.depends_on(depends);
50 h.parallel_for(sycl::range<1>(n), [=](sycl::id<1> i) {
51 TValue left_val = left ? *left : fp[0];
52 TValue right_val = right ? *right : fp[xp_size - 1];
53
54 TCoord x_val = x[i];
55 std::int64_t x_idx = idx[i] - 1;
56
57 if (IsNan<TCoord>::isnan(x_val)) {
58 out[i] = x_val;
59 }
60 else if (x_idx < 0) {
61 out[i] = left_val;
62 }
63 else if (x_val == xp[xp_size - 1]) {
64 out[i] = fp[xp_size - 1];
65 }
66 else if (x_idx >= static_cast<std::int64_t>(xp_size - 1)) {
67 out[i] = right_val;
68 }
69 else {
70 TValue slope =
71 (fp[x_idx + 1] - fp[x_idx]) / (xp[x_idx + 1] - xp[x_idx]);
72 TValue res = slope * (x_val - xp[x_idx]) + fp[x_idx];
73
74 if (IsNan<TValue>::isnan(res)) {
75 res = slope * (x_val - xp[x_idx + 1]) + fp[x_idx + 1];
76 if (IsNan<TValue>::isnan(res) &&
77 (fp[x_idx] == fp[x_idx + 1])) {
78 res = fp[x_idx];
79 }
80 }
81 out[i] = res;
82 }
83 });
84 });
85}
