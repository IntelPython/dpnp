117template <DPNPFuncType FT1, DPNPFuncType FT2>
118static constexpr DPNPFuncType populate_func_types()
119{
120 if constexpr (FT1 == DPNPFuncType::DPNP_FT_NONE) {
121 throw std::runtime_error("Templated enum value of FT1 is None");
122 }
123 else if constexpr (FT2 == DPNPFuncType::DPNP_FT_NONE) {
124 throw std::runtime_error("Templated enum value of FT2 is None");
125 }
126 return (FT1 < FT2) ? FT2 : FT1;
127}

227class dpnp_less_comp
228{
229public:
230 template <typename _Xp, typename _Yp>
231 bool operator()(_Xp &&__x, _Yp &&__y) const
232 {
233 if constexpr (both_types_are_same<
234 dpnp_remove_cvref_t<_Xp>, dpnp_remove_cvref_t<_Yp>,
235 std::complex<float>, std::complex<double>>)
236 {
237 bool ret = false;
238 _Xp a = std::forward<_Xp>(__x);
239 _Yp b = std::forward<_Yp>(__y);
240
241 if (a.real() < b.real()) {
242 ret = (a.imag() == a.imag() || b.imag() != b.imag());
243 }
244 else if (a.real() > b.real()) {
245 ret = (b.imag() != b.imag() && a.imag() == a.imag());
246 }
247 else if (a.real() == b.real() ||
248 (a.real() != a.real() && b.real() != b.real())) {
249 ret = (a.imag() < b.imag() ||
250 (b.imag() != b.imag() && a.imag() == a.imag()));
251 }
252 else {
253 ret = (b.real() != b.real());
254 }
255 return ret;
256 }
257 else {
258 return std::forward<_Xp>(__x) < std::forward<_Yp>(__y);
259 }
260 }
261};

278template <DPNPFuncType FT1,
279 DPNPFuncType FT2,
280 typename has_fp64 = std::true_type,
281 typename keep_int = std::false_type>
282static constexpr DPNPFuncType get_floating_res_type()
283{
284 constexpr auto widest_type = populate_func_types<FT1, FT2>();
285 constexpr auto shortes_type = (widest_type == FT1) ? FT2 : FT1;
286
287 // Return integer result type if save_int is True
288 if constexpr (keep_int::value) {
289 if constexpr (widest_type == DPNPFuncType::DPNP_FT_INT ||
290 widest_type == DPNPFuncType::DPNP_FT_LONG)
291 {
292 return widest_type;
293 }
294 }
295
296 // Check for double
297 if constexpr (widest_type == DPNPFuncType::DPNP_FT_DOUBLE) {
298 return widest_type;
299 }
300
301 // Check for float
302 else if constexpr (widest_type == DPNPFuncType::DPNP_FT_FLOAT) {
303 // Check if the shortest type is also float
304 if constexpr (shortes_type == DPNPFuncType::DPNP_FT_FLOAT) {
305 return widest_type;
306 }
307 }
308
309 // Default case
310 return get_default_floating_type<has_fp64>();
311}
