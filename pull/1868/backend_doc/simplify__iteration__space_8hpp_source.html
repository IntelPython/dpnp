39template <class ShapeTy, class StridesTy>
40int simplify_iteration_two_strides(const int nd,
41 ShapeTy *shape,
42 StridesTy *strides1,
43 StridesTy *strides2,
44 StridesTy &disp1,
45 StridesTy &disp2)
46{
47 disp1 = StridesTy(0);
48 disp2 = StridesTy(0);
49 if (nd < 2)
50 return nd;
51
52 std::vector<int> pos(nd);
53 std::iota(pos.begin(), pos.end(), 0);
54
55 std::stable_sort(
56 pos.begin(), pos.end(), [&strides1, &strides2, &shape](int i1, int i2) {
57 auto abs_str1_i1 =
58 (strides1[i1] < 0) ? -strides1[i1] : strides1[i1];
59 auto abs_str1_i2 =
60 (strides1[i2] < 0) ? -strides1[i2] : strides1[i2];
61 auto abs_str2_i1 =
62 (strides2[i1] < 0) ? -strides2[i1] : strides2[i1];
63 auto abs_str2_i2 =
64 (strides2[i2] < 0) ? -strides2[i2] : strides2[i2];
65 return (abs_str2_i1 > abs_str2_i2) ||
66 (abs_str2_i1 == abs_str2_i2 &&
67 (abs_str1_i1 > abs_str1_i2 ||
68 (abs_str1_i1 == abs_str1_i2 && shape[i1] > shape[i2])));
69 });
70
71 std::vector<ShapeTy> shape_w;
72 std::vector<StridesTy> strides1_w;
73 std::vector<StridesTy> strides2_w;
74
75 bool contractable = true;
76 for (int i = 0; i < nd; ++i) {
77 auto p = pos[i];
78 auto sh_p = shape[p];
79 auto str1_p = strides1[p];
80 auto str2_p = strides2[p];
81 shape_w.push_back(sh_p);
82 if (str1_p <= 0 && str2_p <= 0 && std::min(str1_p, str2_p) < 0) {
83 disp1 += str1_p * (sh_p - 1);
84 str1_p = -str1_p;
85 disp2 += str2_p * (sh_p - 1);
86 str2_p = -str2_p;
87 }
88 if (str1_p < 0 || str2_p < 0) {
89 contractable = false;
90 }
91 strides1_w.push_back(str1_p);
92 strides2_w.push_back(str2_p);
93 }
94
95 int nd_ = nd;
96 while (contractable) {
97 bool changed = false;
98 for (int i = 0; i + 1 < nd_; ++i) {
99 StridesTy str1 = strides1_w[i + 1];
100 StridesTy str2 = strides2_w[i + 1];
101 StridesTy jump1 = strides1_w[i] - (shape_w[i + 1] - 1) * str1;
102 StridesTy jump2 = strides2_w[i] - (shape_w[i + 1] - 1) * str2;
103
104 if (jump1 == str1 && jump2 == str2) {
105 changed = true;
106 shape_w[i] *= shape_w[i + 1];
107 for (int j = i; j < nd_; ++j) {
108 strides1_w[j] = strides1_w[j + 1];
109 }
110 for (int j = i; j < nd_; ++j) {
111 strides2_w[j] = strides2_w[j + 1];
112 }
113 for (int j = i + 1; j + 1 < nd_; ++j) {
114 shape_w[j] = shape_w[j + 1];
115 }
116 --nd_;
117 break;
118 }
119 }
120 if (!changed)
121 break;
122 }
123 for (int i = 0; i < nd_; ++i) {
124 shape[i] = shape_w[i];
125 }
126 for (int i = 0; i < nd_; ++i) {
127 strides1[i] = strides1_w[i];
128 }
129 for (int i = 0; i < nd_; ++i) {
130 strides2[i] = strides2_w[i];
131 }
132
133 return nd_;
134}

136template <class ShapeTy, class StridesTy>
137int simplify_iteration_three_strides(const int nd,
138 ShapeTy *shape,
139 StridesTy *strides1,
140 StridesTy *strides2,
141 StridesTy *strides3,
142 StridesTy &disp1,
143 StridesTy &disp2,
144 StridesTy &disp3)
145{
146 disp1 = StridesTy(0);
147 disp2 = StridesTy(0);
148 if (nd < 2)
149 return nd;
150
151 std::vector<int> pos(nd);
152 std::iota(pos.begin(), pos.end(), 0);
153
154 std::stable_sort(pos.begin(), pos.end(),
155 [&strides1, &strides2, &strides3, &shape](int i1, int i2) {
156 auto abs_str1_i1 =
157 (strides1[i1] < 0) ? -strides1[i1] : strides1[i1];
158 auto abs_str1_i2 =
159 (strides1[i2] < 0) ? -strides1[i2] : strides1[i2];
160 auto abs_str2_i1 =
161 (strides2[i1] < 0) ? -strides2[i1] : strides2[i1];
162 auto abs_str2_i2 =
163 (strides2[i2] < 0) ? -strides2[i2] : strides2[i2];
164 auto abs_str3_i1 =
165 (strides3[i1] < 0) ? -strides3[i1] : strides3[i1];
166 auto abs_str3_i2 =
167 (strides3[i2] < 0) ? -strides3[i2] : strides3[i2];
168 return (abs_str3_i1 > abs_str3_i2) ||
169 ((abs_str3_i1 == abs_str3_i2) &&
170 ((abs_str2_i1 > abs_str2_i2) ||
171 ((abs_str2_i1 == abs_str2_i2) &&
172 ((abs_str1_i1 > abs_str1_i2) ||
173 ((abs_str1_i1 == abs_str1_i2) &&
174 (shape[i1] > shape[i2]))))));
175 });
176
177 std::vector<ShapeTy> shape_w;
178 std::vector<StridesTy> strides1_w;
179 std::vector<StridesTy> strides2_w;
180 std::vector<StridesTy> strides3_w;
181
182 bool contractable = true;
183 for (int i = 0; i < nd; ++i) {
184 auto p = pos[i];
185 auto sh_p = shape[p];
186 auto str1_p = strides1[p];
187 auto str2_p = strides2[p];
188 auto str3_p = strides3[p];
189 shape_w.push_back(sh_p);
190 if (str1_p <= 0 && str2_p <= 0 && str3_p <= 0 &&
191 std::min({str1_p, str2_p, str3_p}) < 0)
192 {
193 disp1 += str1_p * (sh_p - 1);
194 str1_p = -str1_p;
195 disp2 += str2_p * (sh_p - 1);
196 str2_p = -str2_p;
197 disp3 += str3_p * (sh_p - 1);
198 str3_p = -str3_p;
199 }
200 if (str1_p < 0 || str2_p < 0 || str3_p < 0) {
201 contractable = false;
202 }
203 strides1_w.push_back(str1_p);
204 strides2_w.push_back(str2_p);
205 strides3_w.push_back(str3_p);
206 }
207 int nd_ = nd;
208 while (contractable) {
209 bool changed = false;
210 for (int i = 0; i + 1 < nd_; ++i) {
211 StridesTy str1 = strides1_w[i + 1];
212 StridesTy str2 = strides2_w[i + 1];
213 StridesTy str3 = strides3_w[i + 1];
214 StridesTy jump1 = strides1_w[i] - (shape_w[i + 1] - 1) * str1;
215 StridesTy jump2 = strides2_w[i] - (shape_w[i + 1] - 1) * str2;
216 StridesTy jump3 = strides3_w[i] - (shape_w[i + 1] - 1) * str3;
217
218 if (jump1 == str1 && jump2 == str2 && jump3 == str3) {
219 changed = true;
220 shape_w[i] *= shape_w[i + 1];
221 for (int j = i; j < nd_; ++j) {
222 strides1_w[j] = strides1_w[j + 1];
223 }
224 for (int j = i; j < nd_; ++j) {
225 strides2_w[j] = strides2_w[j + 1];
226 }
227 for (int j = i; j < nd_; ++j) {
228 strides3_w[j] = strides3_w[j + 1];
229 }
230 for (int j = i + 1; j + 1 < nd_; ++j) {
231 shape_w[j] = shape_w[j + 1];
232 }
233 --nd_;
234 break;
235 }
236 }
237 if (!changed)
238 break;
239 }
240 for (int i = 0; i < nd_; ++i) {
241 shape[i] = shape_w[i];
242 }
243 for (int i = 0; i < nd_; ++i) {
244 strides1[i] = strides1_w[i];
245 }
246 for (int i = 0; i < nd_; ++i) {
247 strides2[i] = strides2_w[i];
248 }
249 for (int i = 0; i < nd_; ++i) {
250 strides3[i] = strides3_w[i];
251 }
252
253 return nd_;
254}

256void simplify_iteration_space(int &nd,
257 const py::ssize_t *const &shape,
258 std::vector<py::ssize_t> const &src_strides,
259 std::vector<py::ssize_t> const &dst_strides,
260 // output
261 std::vector<py::ssize_t> &simplified_shape,
262 std::vector<py::ssize_t> &simplified_src_strides,
263 std::vector<py::ssize_t> &simplified_dst_strides,
264 py::ssize_t &src_offset,
265 py::ssize_t &dst_offset)
266{
267 if (nd > 1) {
268 // Simplify iteration space to reduce dimensionality
269 // and improve access pattern
270 simplified_shape.reserve(nd);
271 simplified_shape.insert(std::begin(simplified_shape), shape,
272 shape + nd);
273 assert(simplified_shape.size() == static_cast<size_t>(nd));
274
275 simplified_src_strides.reserve(nd);
276 simplified_src_strides.insert(std::end(simplified_src_strides),
277 std::begin(src_strides),
278 std::end(src_strides));
279 assert(simplified_src_strides.size() == static_cast<size_t>(nd));
280
281 simplified_dst_strides.reserve(nd);
282 simplified_dst_strides.insert(std::end(simplified_dst_strides),
283 std::begin(dst_strides),
284 std::end(dst_strides));
285 assert(simplified_dst_strides.size() == static_cast<size_t>(nd));
286
287 int contracted_nd = simplify_iteration_two_strides(
288 nd, simplified_shape.data(), simplified_src_strides.data(),
289 simplified_dst_strides.data(),
290 src_offset, // modified by reference
291 dst_offset // modified by reference
292 );
293 simplified_shape.resize(contracted_nd);
294 simplified_src_strides.resize(contracted_nd);
295 simplified_dst_strides.resize(contracted_nd);
296
297 nd = contracted_nd;
298 }
299 else if (nd == 1) {
300 src_offset = 0;
301 dst_offset = 0;
302 // Populate vectors
303 simplified_shape.reserve(nd);
304 simplified_shape.push_back(shape[0]);
305 assert(simplified_shape.size() == static_cast<size_t>(nd));
306
307 simplified_src_strides.reserve(nd);
308 simplified_dst_strides.reserve(nd);
309
310 if (src_strides[0] < 0 && dst_strides[0] < 0) {
311 simplified_src_strides.push_back(-src_strides[0]);
312 simplified_dst_strides.push_back(-dst_strides[0]);
313 if (shape[0] > 1) {
314 src_offset += (shape[0] - 1) * src_strides[0];
315 dst_offset += (shape[0] - 1) * dst_strides[0];
316 }
317 }
318 else {
319 simplified_src_strides.push_back(src_strides[0]);
320 simplified_dst_strides.push_back(dst_strides[0]);
321 }
322
323 assert(simplified_src_strides.size() == static_cast<size_t>(nd));
324 assert(simplified_dst_strides.size() == static_cast<size_t>(nd));
325 }
326}

328void simplify_iteration_space_3(
329 int &nd,
330 const py::ssize_t *const &shape,
331 // src1
332 std::vector<py::ssize_t> const &src1_strides,
333 // src2
334 std::vector<py::ssize_t> const &src2_strides,
335 // dst
336 std::vector<py::ssize_t> const &dst_strides,
337 // output
338 std::vector<py::ssize_t> &simplified_shape,
339 std::vector<py::ssize_t> &simplified_src1_strides,
340 std::vector<py::ssize_t> &simplified_src2_strides,
341 std::vector<py::ssize_t> &simplified_dst_strides,
342 py::ssize_t &src1_offset,
343 py::ssize_t &src2_offset,
344 py::ssize_t &dst_offset)
345{
346 if (nd > 1) {
347 // Simplify iteration space to reduce dimensionality
348 // and improve access pattern
349 simplified_shape.reserve(nd);
350 simplified_shape.insert(std::end(simplified_shape), shape, shape + nd);
351 assert(simplified_shape.size() == static_cast<size_t>(nd));
352
353 simplified_src1_strides.reserve(nd);
354 simplified_src1_strides.insert(std::end(simplified_src1_strides),
355 std::begin(src1_strides),
356 std::end(src1_strides));
357 assert(simplified_src1_strides.size() == static_cast<size_t>(nd));
358
359 simplified_src2_strides.reserve(nd);
360 simplified_src2_strides.insert(std::end(simplified_src2_strides),
361 std::begin(src2_strides),
362 std::end(src2_strides));
363 assert(simplified_src2_strides.size() == static_cast<size_t>(nd));
364
365 simplified_dst_strides.reserve(nd);
366 simplified_dst_strides.insert(std::end(simplified_dst_strides),
367 std::begin(dst_strides),
368 std::end(dst_strides));
369 assert(simplified_dst_strides.size() == static_cast<size_t>(nd));
370
371 int contracted_nd = simplify_iteration_three_strides(
372 nd, simplified_shape.data(), simplified_src1_strides.data(),
373 simplified_src2_strides.data(), simplified_dst_strides.data(),
374 src1_offset, // modified by reference
375 src2_offset, // modified by reference
376 dst_offset // modified by reference
377 );
378 simplified_shape.resize(contracted_nd);
379 simplified_src1_strides.resize(contracted_nd);
380 simplified_src2_strides.resize(contracted_nd);
381 simplified_dst_strides.resize(contracted_nd);
382
383 nd = contracted_nd;
384 }
385 else if (nd == 1) {
386 src1_offset = 0;
387 src2_offset = 0;
388 dst_offset = 0;
389 // Populate vectors
390 simplified_shape.reserve(nd);
391 simplified_shape.push_back(shape[0]);
392 assert(simplified_shape.size() == static_cast<size_t>(nd));
393
394 simplified_src1_strides.reserve(nd);
395 simplified_src2_strides.reserve(nd);
396 simplified_dst_strides.reserve(nd);
397
398 if ((src1_strides[0] < 0) && (src2_strides[0] < 0) &&
399 (dst_strides[0] < 0)) {
400 simplified_src1_strides.push_back(-src1_strides[0]);
401 simplified_src2_strides.push_back(-src2_strides[0]);
402 simplified_dst_strides.push_back(-dst_strides[0]);
403 if (shape[0] > 1) {
404 src1_offset += src1_strides[0] * (shape[0] - 1);
405 src2_offset += src2_strides[0] * (shape[0] - 1);
406 dst_offset += dst_strides[0] * (shape[0] - 1);
407 }
408 }
409 else {
410 simplified_src1_strides.push_back(src1_strides[0]);
411 simplified_src2_strides.push_back(src2_strides[0]);
412 simplified_dst_strides.push_back(dst_strides[0]);
413 }
414
415 assert(simplified_src1_strides.size() == static_cast<size_t>(nd));
416 assert(simplified_src2_strides.size() == static_cast<size_t>(nd));
417 assert(simplified_dst_strides.size() == static_cast<size_t>(nd));
418 }
