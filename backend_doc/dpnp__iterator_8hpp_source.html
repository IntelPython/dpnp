155  inline pointer ptr(size_type iteration_id) const
156  {
157  size_type offset = 0;
158 
159  if (iteration_shape_size > 0)
160  {
161  long reminder = iteration_id;
162  for (size_t it = 0; it < iteration_shape_size; ++it)
163  {
164  const size_type axis_val = iteration_shape_strides[it];
165  size_type xyz_id = reminder / axis_val;
166  offset += (xyz_id * axes_shape_strides[it]);
167 
168  reminder = reminder % axis_val;
169  }
170  }
171  else
172  {
173  offset = iteration_id;
174  }
175 
176  return base + offset;
177  }
178

181  const size_type iteration_shape_size = size_type{};
182  const size_type* iteration_shape_strides = nullptr;

251  inline void broadcast_to_shape(const std::vector<size_type>& __shape)
252  {
253  if (axis_use)
254  {
255  return;
256  }
257 
258  if (broadcastable(input_shape, input_shape_size, __shape))
259  {
260  free_broadcast_axes_memory();
261  free_output_memory();
262 
263  std::vector<size_type> valid_axes;
264  broadcast_use = true;
265 
266  output_shape_size = __shape.size();
267  const size_type output_shape_size_in_bytes = output_shape_size * sizeof(size_type);
268  output_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
269 
270  for (int irit = input_shape_size - 1, orit = output_shape_size - 1; orit >= 0; --irit, --orit)
271  {
272  output_shape[orit] = __shape[orit];
273 
274  // ex: input_shape = {7, 1, 5}, output_shape = {8, 7, 6, 5} => valid_axes = {0, 2}
275  if (irit < 0 || input_shape[irit] != output_shape[orit])
276  {
277  valid_axes.insert(valid_axes.begin(), orit);
278  }
279  }
280 
281  broadcast_axes_size = valid_axes.size();
282  const size_type broadcast_axes_size_in_bytes = broadcast_axes_size * sizeof(size_type);
283  broadcast_axes = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(broadcast_axes_size_in_bytes));
284  std::copy(valid_axes.begin(), valid_axes.end(), broadcast_axes);
285 
286  output_size = std::accumulate(
287  output_shape, output_shape + output_shape_size, size_type(1), std::multiplies<size_type>());
288 
289  output_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
290  get_shape_offsets_inkernel<size_type>(output_shape, output_shape_size, output_shape_strides);
291 
292  iteration_size = 1;
293  }
294  }

341  inline void set_axes(const std::vector<long>& __axes)
342  {
343  if (broadcast_use)
344  {
345  return;
346  }
347 
348  if (!__axes.empty() && input_shape_size)
349  {
350  free_axes_memory();
351  free_iteration_memory();
352  free_output_memory();
353 
354  axes = get_validated_axes(__axes, input_shape_size);
355  axis_use = true;
356 
357  output_shape_size = input_shape_size - axes.size();
358  const size_type output_shape_size_in_bytes = output_shape_size * sizeof(size_type);
359 
360  iteration_shape_size = axes.size();
361  const size_type iteration_shape_size_in_bytes = iteration_shape_size * sizeof(size_type);
362  std::vector<size_type> iteration_shape;
363 
364  output_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
365  size_type* output_shape_it = output_shape;
366  for (size_type i = 0; i < input_shape_size; ++i)
367  {
368  if (std::find(axes.begin(), axes.end(), i) == axes.end())
369  {
370  *output_shape_it = input_shape[i];
371  ++output_shape_it;
372  }
373  }
374 
375  output_size = std::accumulate(
376  output_shape, output_shape + output_shape_size, size_type(1), std::multiplies<size_type>());
377 
378  output_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
379  get_shape_offsets_inkernel<size_type>(output_shape, output_shape_size, output_shape_strides);
380 
381  iteration_size = 1;
382  iteration_shape.reserve(iteration_shape_size);
383  for (const auto& axis : axes)
384  {
385  const size_type axis_dim = input_shape[axis];
386  iteration_shape.push_back(axis_dim);
387  iteration_size *= axis_dim;
388  }
389 
390  iteration_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(iteration_shape_size_in_bytes));
391  get_shape_offsets_inkernel<size_type>(
392  iteration_shape.data(), iteration_shape.size(), iteration_shape_strides);
393 
394  axes_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(iteration_shape_size_in_bytes));
395  for (size_t i = 0; i < iteration_shape_size; ++i)
396  {
397  axes_shape_strides[i] = input_shape_strides[axes[i]];
398  }
399  }
400  }

437  void init_container(pointer __ptr, const std::vector<size_type>& __shape)
438  {
439  // TODO needs to address negative values in __shape with exception
440  if ((__ptr == nullptr) && __shape.empty())
441  {
442  return;
443  }
444 
445  if (__ptr != nullptr)
446  {
447  data = __ptr;
448  input_size = 1; // means scalar at this stage
449  output_size = 1; // if input size is not zero it means we have scalar as output
450  iteration_size = 1;
451  }
452 
453  if (!__shape.empty())
454  {
455  input_size = std::accumulate(__shape.begin(), __shape.end(), size_type(1), std::multiplies<size_type>());
456  if (input_size == 0)
457  { // shape might be shape[3, 4, 0, 6]. This means no input memory and no output expected
458  output_size = 0; // depends on axes. zero at this stage only
459  }
460 
461  input_shape_size = __shape.size();
462  input_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(input_shape_size * sizeof(size_type)));
463  std::copy(__shape.begin(), __shape.end(), input_shape);
464 
465  input_shape_strides =
466  reinterpret_cast<size_type*>(dpnp_memory_alloc_c(input_shape_size * sizeof(size_type)));
467  get_shape_offsets_inkernel<size_type>(input_shape, input_shape_size, input_shape_strides);
468  }
469  iteration_size = input_size;
470  }

473  size_type get_input_begin_offset(size_type output_global_id) const
474  {
475  size_type input_global_id = 0;
476  if (axis_use)
477  {
478  assert(output_global_id < output_size);
479 
480  for (size_t iit = 0, oit = 0; iit < input_shape_size; ++iit)
481  {
482  if (std::find(axes.begin(), axes.end(), iit) == axes.end())
483  {
484  const size_type output_xyz_id =
485  get_xyz_id_by_id_inkernel(output_global_id, output_shape_strides, output_shape_size, oit);
486  input_global_id += (output_xyz_id * input_shape_strides[iit]);
487  ++oit;
488  }
489  }
490  }
491  else if (broadcast_use)
492  {
493  assert(output_global_id < output_size);
494  assert(input_shape_size <= output_shape_size);
495 
496  for (int irit = input_shape_size - 1, orit = output_shape_size - 1; irit >= 0; --irit, --orit)
497  {
498  size_type* broadcast_axes_end = broadcast_axes + broadcast_axes_size;
499  if (std::find(broadcast_axes, broadcast_axes_end, orit) == broadcast_axes_end)
500  {
501  const size_type output_xyz_id =
502  get_xyz_id_by_id_inkernel(output_global_id, output_shape_strides, output_shape_size, orit);
503  input_global_id += (output_xyz_id * input_shape_strides[irit]);
504  }
505  }
506  }
507 
508  return input_global_id;
509  }
