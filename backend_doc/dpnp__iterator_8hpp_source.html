169  DPNPC_id(void* __ptr, const std::vector<size_type>& __shape)
170  {
171  data = reinterpret_cast<pointer>(__ptr);
172  if (!__shape.empty())
173  {
174  shape_size = __shape.size();
175  shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(shape_size * sizeof(size_type)));
176  std::copy(__shape.begin(), __shape.end(), shape);
177 
178  shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(shape_size * sizeof(size_type)));
179  get_shape_offsets_inkernel<size_type>(shape, shape_size, shape_strides);
180 
181  size = std::accumulate(__shape.begin(), __shape.end(), size_type(1), std::multiplies<size_type>());
182  }
183  }
184

190  dpnp_memory_free_c(shape_strides);
191  dpnp_memory_free_c(output_shape);
192  dpnp_memory_free_c(output_shape_strides);
193  dpnp_memory_free_c(sycl_output_xyz);
194  dpnp_memory_free_c(sycl_input_xyz);
195  }

205  inline void set_axis(size_type __axis)
206  {
207  if (__axis < shape_size)
208  {
209  axis = __axis;
210  axis_use = true;
211 
212  output_shape_size = shape_size - 1;
213  const size_type output_shape_size_in_bytes = output_shape_size * sizeof(size_type);
214 
215  output_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
216  size_type* output_shape_it = std::copy(shape, shape + axis, output_shape);
217  std::copy(shape + axis + 1, shape + shape_size, output_shape_it);
218 
219  output_size = std::accumulate(
220  output_shape, output_shape + output_shape_size, size_type(1), std::multiplies<size_type>());
221 
222  output_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
223  get_shape_offsets_inkernel<size_type>(output_shape, output_shape_size, output_shape_strides);
224 
225  // make thread private storage for each shape by multiplying memory
226  sycl_output_xyz =
227  reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_size * output_shape_size_in_bytes));
228  sycl_input_xyz =
229  reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_size * shape_size * sizeof(size_type)));
230  }

258  size_type get_input_begin_offset(size_type output_global_id) const
259  {
260  size_type input_global_id = 0;
261  if (axis_use)
262  {
263  assert(output_global_id < output_size);
264 
265  // use thread private storage
266  size_type* sycl_output_xyz_thread = sycl_output_xyz + (output_global_id * output_shape_size);
267  size_type* sycl_input_xyz_thread = sycl_input_xyz + (output_global_id * shape_size);
268 
269  get_xyz_by_id_inkernel(output_global_id, output_shape_strides, output_shape_size, sycl_output_xyz_thread);
270 
271  for (size_t iit = 0, oit = 0; iit < shape_size; ++iit)
272  {
273  if (iit == axis)
274  {
275  sycl_input_xyz_thread[iit] = 0; // put begin point of the axis
276  }
277  else
278  {
279  sycl_input_xyz_thread[iit] = sycl_output_xyz_thread[oit];
280  ++oit;
281  }
282  }
283 
284  input_global_id = get_id_by_xyz_inkernel(sycl_input_xyz_thread, shape_size, shape_strides);
285  }
286 
287  return input_global_id;
288  }
