155  inline pointer ptr(size_type iteration_id) const
156  {
157  size_type offset = 0;
158 
159  if (iteration_shape_size > 0)
160  {
161  long reminder = iteration_id;
162  for (size_t it = 0; it < iteration_shape_size; ++it)
163  {
164  const size_type axis_val = iteration_shape_strides[it];
165  size_type xyz_id = reminder / axis_val;
166  offset += (xyz_id * axes_shape_strides[it]);
167 
168  reminder = reminder % axis_val;
169  }
170  }
171  else
172  {
173  offset = iteration_id;
174  }
175 
176  return base + offset;
177  }

251  inline void broadcast_to_shape(const std::vector<size_type>& __shape)
252  {
253  if (axis_use)
254  {
255  return;
256  }
257 
258  if (broadcastable(input_shape, input_shape_size, __shape))
259  {
260  free_broadcast_axes_memory();
261  free_output_memory();
262 
263  std::vector<size_type> valid_axes;
264  broadcast_use = true;
265 
266  output_shape_size = __shape.size();
267  const size_type output_shape_size_in_bytes = output_shape_size * sizeof(size_type);
268  output_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
269 
270  for (int irit = input_shape_size - 1, orit = output_shape_size - 1; orit >= 0; --irit, --orit)
271  {
272  output_shape[orit] = __shape[orit];
273 
274  // ex: input_shape = {7, 1, 5}, output_shape = {8, 7, 6, 5} => valid_axes = {0, 2}
275  if (irit < 0 || input_shape[irit] != output_shape[orit])
276  {
277  valid_axes.insert(valid_axes.begin(), orit);
278  }
279  }
280 
281  broadcast_axes_size = valid_axes.size();
282  const size_type broadcast_axes_size_in_bytes = broadcast_axes_size * sizeof(size_type);
283  broadcast_axes = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(broadcast_axes_size_in_bytes));
284  std::copy(valid_axes.begin(), valid_axes.end(), broadcast_axes);
285 
286  output_size = std::accumulate(
287  output_shape, output_shape + output_shape_size, size_type(1), std::multiplies<size_type>());
288 
289  output_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
290  get_shape_offsets_inkernel<size_type>(output_shape, output_shape_size, output_shape_strides);
291 
292  iteration_size = 1;
293 
294  // make thread private storage for each shape by multiplying memory
295  sycl_output_xyz =
296  reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_size * output_shape_size_in_bytes));
297  }
298  }

345  inline void set_axes(const std::vector<long>& __axes)
346  {
347  if (broadcast_use)
348  {
349  return;
350  }
351 
352  if (!__axes.empty() && input_shape_size)
353  {
354  free_axes_memory();
355  free_iteration_memory();
356  free_output_memory();
357 
358  axes = get_validated_axes(__axes, input_shape_size);
359  axis_use = true;
360 
361  output_shape_size = input_shape_size - axes.size();
362  const size_type output_shape_size_in_bytes = output_shape_size * sizeof(size_type);
363 
364  iteration_shape_size = axes.size();
365  const size_type iteration_shape_size_in_bytes = iteration_shape_size * sizeof(size_type);
366  std::vector<size_type> iteration_shape;
367 
368  output_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
369  size_type* output_shape_it = output_shape;
370  for (size_type i = 0; i < input_shape_size; ++i)
371  {
372  if (std::find(axes.begin(), axes.end(), i) == axes.end())
373  {
374  *output_shape_it = input_shape[i];
375  ++output_shape_it;
376  }
377  }
378 
379  output_size = std::accumulate(
380  output_shape, output_shape + output_shape_size, size_type(1), std::multiplies<size_type>());
381 
382  output_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
383  get_shape_offsets_inkernel<size_type>(output_shape, output_shape_size, output_shape_strides);
384 
385  iteration_size = 1;
386  iteration_shape.reserve(iteration_shape_size);
387  for (const auto& axis : axes)
388  {
389  const size_type axis_dim = input_shape[axis];
390  iteration_shape.push_back(axis_dim);
391  iteration_size *= axis_dim;
392  }
393 
394  iteration_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(iteration_shape_size_in_bytes));
395  get_shape_offsets_inkernel<size_type>(
396  iteration_shape.data(), iteration_shape.size(), iteration_shape_strides);
397 
398  axes_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(iteration_shape_size_in_bytes));
399  for (size_t i = 0; i < iteration_shape_size; ++i)
400  {
401  axes_shape_strides[i] = input_shape_strides[axes[i]];
402  }
403 
404  // make thread private storage for each shape by multiplying memory
405  sycl_output_xyz =
406  reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_size * output_shape_size_in_bytes));
407  }
408  }

445  void init_container(pointer __ptr, const std::vector<size_type>& __shape)
446  {
447  // TODO needs to address negative values in __shape with exception
448  if ((__ptr == nullptr) && __shape.empty())
449  {
450  return;
451  }
452 
453  if (__ptr != nullptr)
454  {
455  data = __ptr;
456  input_size = 1; // means scalar at this stage
457  output_size = 1; // if input size is not zero it means we have scalar as output
458  iteration_size = 1;
459  }
460 
461  if (!__shape.empty())
462  {
463  input_size = std::accumulate(__shape.begin(), __shape.end(), size_type(1), std::multiplies<size_type>());
464  if (input_size == 0)
465  { // shape might be shape[3, 4, 0, 6]. This means no input memory and no output expected
466  output_size = 0; // depends on axes. zero at this stage only
467  }
468 
469  input_shape_size = __shape.size();
470  input_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(input_shape_size * sizeof(size_type)));
471  std::copy(__shape.begin(), __shape.end(), input_shape);
472 
473  input_shape_strides =
474  reinterpret_cast<size_type*>(dpnp_memory_alloc_c(input_shape_size * sizeof(size_type)));
475  get_shape_offsets_inkernel<size_type>(input_shape, input_shape_size, input_shape_strides);
476  }
477  iteration_size = input_size;
478  }

481  size_type get_input_begin_offset(size_type output_global_id) const
482  {
483  size_type input_global_id = 0;
484  if (axis_use)
485  {
486  assert(output_global_id < output_size);
487 
488  // use thread private storage
489  size_type* sycl_output_xyz_thread = sycl_output_xyz + (output_global_id * output_shape_size);
490 
491  get_xyz_by_id_inkernel(output_global_id, output_shape_strides, output_shape_size, sycl_output_xyz_thread);
492 
493  for (size_t iit = 0, oit = 0; iit < input_shape_size; ++iit)
494  {
495  if (std::find(axes.begin(), axes.end(), iit) == axes.end())
496  {
497  input_global_id += (sycl_output_xyz_thread[oit] * input_shape_strides[iit]);
498  ++oit;
499  }
500  }
501  }
502  else if (broadcast_use)
503  {
504  assert(output_global_id < output_size);
505 
506  // use thread private storage
507  size_type* sycl_output_xyz_thread = sycl_output_xyz + (output_global_id * output_shape_size);
508 
509  get_xyz_by_id_inkernel(output_global_id, output_shape_strides, output_shape_size, sycl_output_xyz_thread);
510 
511  for (int irit = input_shape_size - 1, orit = output_shape_size - 1; irit >= 0; --irit, --orit)
512  {
513  size_type* broadcast_axes_end = broadcast_axes + broadcast_axes_size;
514  if (std::find(broadcast_axes, broadcast_axes_end, orit) == broadcast_axes_end)
515  {
516  input_global_id += (sycl_output_xyz_thread[orit] * input_shape_strides[irit]);
517  }
518  }
519  }
520 
521  return input_global_id;
522  }
