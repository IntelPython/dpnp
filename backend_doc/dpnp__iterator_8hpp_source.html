155 inline pointer ptr(size_type iteration_id) const
156 {
157 size_type offset = 0;
158
159 if (iteration_shape_size > 0)
160 {
161 long reminder = iteration_id;
162 for (size_t it = 0; it < iteration_shape_size; ++it)
163 {
164 const size_type axis_val = iteration_shape_strides[it];
165 size_type xyz_id = reminder / axis_val;
166 offset += (xyz_id * axes_shape_strides[it]);
167
168 reminder = reminder % axis_val;
169 }
170 }
171 else
172 {
173 offset = iteration_id;
174 }
175
176 return base + offset;
177 }

181 const size_type iteration_shape_size = size_type{};
182 const size_type* iteration_shape_strides = nullptr;
183 const size_type* axes_shape_strides = nullptr;

290 inline void broadcast_to_shape(const std::vector<size_type>& __shape)
291 {
292 if (axis_use)
293 {
294 return;
295 }
296
297 if (broadcastable(input_shape, input_shape_size, __shape))
298 {
299 free_broadcast_axes_memory();
300 free_output_memory();
301
302 std::vector<size_type> valid_axes;
303 broadcast_use = true;
304
305 output_shape_size = __shape.size();
306 const size_type output_shape_size_in_bytes = output_shape_size * sizeof(size_type);
307 output_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(queue_ref, output_shape_size_in_bytes));
308
309 for (int irit = input_shape_size - 1, orit = output_shape_size - 1; orit >= 0; --irit, --orit)
310 {
311 output_shape[orit] = __shape[orit];
312
313 // ex: input_shape = {7, 1, 5}, output_shape = {8, 7, 6, 5} => valid_axes = {0, 2}
314 if (irit < 0 || input_shape[irit] != output_shape[orit])
315 {
316 valid_axes.insert(valid_axes.begin(), orit);
317 }
318 }
319
320 broadcast_axes_size = valid_axes.size();
321 const size_type broadcast_axes_size_in_bytes = broadcast_axes_size * sizeof(size_type);
322 broadcast_axes = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(queue_ref,
323 broadcast_axes_size_in_bytes));
324 std::copy(valid_axes.begin(), valid_axes.end(), broadcast_axes);
325
326 output_size = std::accumulate(
327 output_shape, output_shape + output_shape_size, size_type(1), std::multiplies<size_type>());
328
329 output_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(queue_ref,
330 output_shape_size_in_bytes));
331 get_shape_offsets_inkernel<size_type>(output_shape, output_shape_size, output_shape_strides);
332
333 iteration_size = 1;
334 }
335 }

382 inline void set_axes(const std::vector<shape_elem_type>& __axes)
383 {
384 if (broadcast_use)
385 {
386 return;
387 }
388
389 if (!__axes.empty() && input_shape_size)
390 {
391 free_axes_memory();
392 free_iteration_memory();
393 free_output_memory();
394
395 axes = get_validated_axes(__axes, input_shape_size);
396 axis_use = true;
397
398 output_shape_size = input_shape_size - axes.size();
399 const size_type output_shape_size_in_bytes = output_shape_size * sizeof(size_type);
400
401 iteration_shape_size = axes.size();
402 const size_type iteration_shape_size_in_bytes = iteration_shape_size * sizeof(size_type);
403 std::vector<size_type> iteration_shape;
404
405 output_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(queue_ref, output_shape_size_in_bytes));
406 size_type* output_shape_it = output_shape;
407 for (size_type i = 0; i < input_shape_size; ++i)
408 {
409 if (std::find(axes.begin(), axes.end(), i) == axes.end())
410 {
411 *output_shape_it = input_shape[i];
412 ++output_shape_it;
413 }
414 }
415
416 output_size = std::accumulate(
417 output_shape, output_shape + output_shape_size, size_type(1), std::multiplies<size_type>());
418
419 output_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(queue_ref,
420 output_shape_size_in_bytes));
421 get_shape_offsets_inkernel<size_type>(output_shape, output_shape_size, output_shape_strides);
422
423 iteration_size = 1;
424 iteration_shape.reserve(iteration_shape_size);
425 for (const auto& axis : axes)
426 {
427 const size_type axis_dim = input_shape[axis];
428 iteration_shape.push_back(axis_dim);
429 iteration_size *= axis_dim;
430 }
431
432 iteration_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(queue_ref,
433 iteration_shape_size_in_bytes));
434 get_shape_offsets_inkernel<size_type>(
435 iteration_shape.data(), iteration_shape.size(), iteration_shape_strides);
436
437 axes_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(queue_ref,
438 iteration_shape_size_in_bytes));
439 for (size_t i = 0; i < iteration_shape_size; ++i)
440 {
441 axes_shape_strides[i] = input_shape_strides[axes[i]];
442 }
443 }
444 }

481 void init_container(pointer __ptr, const std::vector<size_type>& __shape)
482 {
483 // TODO needs to address negative values in __shape with exception
484 if ((__ptr == nullptr) && __shape.empty())
485 {
486 return;
487 }
488
489 if (__ptr != nullptr)
490 {
491 data = __ptr;
492 input_size = 1; // means scalar at this stage
493 output_size = 1; // if input size is not zero it means we have scalar as output
494 iteration_size = 1;
495 }
496
497 if (!__shape.empty())
498 {
499 input_size = std::accumulate(__shape.begin(), __shape.end(), size_type(1), std::multiplies<size_type>());
500 if (input_size == 0)
501 { // shape might be shape[3, 4, 0, 6]. This means no input memory and no output expected
502 output_size = 0; // depends on axes. zero at this stage only
503 }
504
505 input_shape_size = __shape.size();
506 input_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(queue_ref,
507 input_shape_size * sizeof(size_type)));
508 std::copy(__shape.begin(), __shape.end(), input_shape);
509
510 input_shape_strides =
511 reinterpret_cast<size_type*>(dpnp_memory_alloc_c(queue_ref, input_shape_size * sizeof(size_type)));
512 get_shape_offsets_inkernel<size_type>(input_shape, input_shape_size, input_shape_strides);
513 }
514 iteration_size = input_size;
515 }

554 size_type get_input_begin_offset(size_type output_global_id) const
555 {
556 size_type input_global_id = 0;
557 if (axis_use)
558 {
559 assert(output_global_id < output_size);
560
561 for (size_t iit = 0, oit = 0; iit < input_shape_size; ++iit)
562 {
563 if (std::find(axes.begin(), axes.end(), iit) == axes.end())
564 {
565 const size_type output_xyz_id =
566 get_xyz_id_by_id_inkernel(output_global_id, output_shape_strides, output_shape_size, oit);
567 input_global_id += (output_xyz_id * input_shape_strides[iit]);
568 ++oit;
569 }
570 }
571 }
572 else if (broadcast_use)
573 {
574 assert(output_global_id < output_size);
575 assert(input_shape_size <= output_shape_size);
576
577 for (int irit = input_shape_size - 1, orit = output_shape_size - 1; irit >= 0; --irit, --orit)
578 {
579 size_type* broadcast_axes_end = broadcast_axes + broadcast_axes_size;
580 if (std::find(broadcast_axes, broadcast_axes_end, orit) == broadcast_axes_end)
581 {
582 const size_type output_xyz_id =
583 get_xyz_id_by_id_inkernel(output_global_id, output_shape_strides, output_shape_size, orit);
584 input_global_id += (output_xyz_id * input_shape_strides[irit]);
585 }
586 }
587 }
588
589 return input_global_id;
590 }
