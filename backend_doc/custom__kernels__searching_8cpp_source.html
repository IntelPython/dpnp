36 void custom_argmax_c(void* array1_in, void* result1, size_t size)
37 {
38  _DataType* array_1 = reinterpret_cast<_DataType*>(array1_in);
39  _idx_DataType* result = reinterpret_cast<_idx_DataType*>(result1);
40 
41  auto policy =
42  oneapi::dpl::execution::make_device_policy<class custom_argmax_c_kernel<_DataType, _idx_DataType>>(DPNP_QUEUE);
43 
44  _DataType* res = std::max_element(policy, array_1, array_1 + size);
