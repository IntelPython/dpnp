79template <typename _DataType>
80void get_shape_offsets_inkernel(const _DataType* shape, size_t shape_size, _DataType* offsets)
81{
82 size_t dim_prod_input = 1;
83 for (size_t i = 0; i < shape_size; ++i)
84 {
85 long i_reverse = shape_size - 1 - i;
86 offsets[i_reverse] = dim_prod_input;
87 dim_prod_input *= shape[i_reverse];
88 }
89
90 return;
91}

107template <typename _DataType>
108void get_xyz_by_id(size_t idx, size_t ndim, const _DataType* offsets, _DataType* xyz)
109{
110 size_t quotient;
111 size_t remainder = idx;
112
113 for (size_t i = 0; i < ndim; ++i)
114 {
115 quotient = remainder / offsets[i];
116 remainder = remainder - quotient * offsets[i];
117 xyz[i] = quotient;
118 }
119 return;
120}

138template <typename _DataType>
139_DataType get_xyz_id_by_id_inkernel(size_t global_id, const _DataType* offsets, size_t offsets_size, size_t axis)
140{
141 /* avoid warning unused variable*/
142 (void)offsets_size;
143
144 assert(axis < offsets_size);
145
146 _DataType xyz_id = 0;
147 long reminder = global_id;
148 for (size_t i = 0; i < axis + 1; ++i)
149 {
150 const _DataType axis_val = offsets[i];
151 xyz_id = reminder / axis_val;
152 reminder = reminder % axis_val;
153 }
154
155 return xyz_id;
156}

173template <typename _DataType>
174size_t get_id_by_xyz_inkernel(const _DataType* xyz, size_t xyz_size, const _DataType* offsets)
175{
176 size_t global_id = 0;
177
178 /* calculate linear index based on reconstructed [x][y][z] */
179 for (size_t it = 0; it < xyz_size; ++it)
180 {
181 global_id += (xyz[it] * offsets[it]);
182 }
183
184 return global_id;
185}

196static inline bool broadcastable(const std::vector<shape_elem_type>& input_shape,
197 const std::vector<shape_elem_type>& output_shape)
198{
199 if (input_shape.size() > output_shape.size())
200 {
201 return false;
202 }
203
204 std::vector<shape_elem_type>::const_reverse_iterator irit = input_shape.rbegin();
205 std::vector<shape_elem_type>::const_reverse_iterator orit = output_shape.rbegin();
206 for (; irit != input_shape.rend(); ++irit, ++orit)
207 {
208 if (*irit != 1 && *irit != *orit)
209 {
210 return false;
211 }
212 }
213
214 return true;
215}

293template <typename _DataType>
294static inline std::vector<_DataType> get_result_shape(const _DataType* input1_shape,
295 const size_t input1_shape_size,
296 const _DataType* input2_shape,
297 const size_t input2_shape_size)
298{
299 const size_t result_shape_size = (input2_shape_size > input1_shape_size) ? input2_shape_size : input1_shape_size;
300 std::vector<_DataType> result_shape;
301 result_shape.reserve(result_shape_size);
302
303 for (int irit1 = input1_shape_size - 1, irit2 = input2_shape_size - 1; irit1 >= 0 || irit2 >= 0; --irit1, --irit2)
304 {
305 _DataType input1_val = (irit1 >= 0) ? input1_shape[irit1] : 1;
306 _DataType input2_val = (irit2 >= 0) ? input2_shape[irit2] : 1;
307
308 if (input1_val == input2_val || input1_val == 1)
309 {
310 result_shape.insert(result_shape.begin(), input2_val);
311 }
312 else if (input2_val == 1)
313 {
314 result_shape.insert(result_shape.begin(), input1_val);
315 }
316 else
317 {
318 throw std::domain_error("DPNP Error: get_common_shape() failed with input shapes check");
319 }
320 }
321
322 return result_shape;
323}

340static inline std::vector<shape_elem_type> get_validated_axes(const std::vector<shape_elem_type>& __axes,
341 const size_t __shape_size,
342 const bool __allow_duplicate = false)
343{
344 std::vector<shape_elem_type> result;
345
346 if (__axes.empty())
347 {
348 goto out;
349 }
350
351 if (__axes.size() > __shape_size)
352 {
353 goto err;
354 }
355
356 result.reserve(__axes.size());
357 for (std::vector<shape_elem_type>::const_iterator it = __axes.cbegin(); it != __axes.cend(); ++it)
358 {
359 const shape_elem_type _axis = *it;
360 const shape_elem_type input_shape_size_signed = static_cast<shape_elem_type>(__shape_size);
361 if (_axis >= input_shape_size_signed)
362 { // positive axis range check
363 goto err;
364 }
365
366 if (_axis < -input_shape_size_signed)
367 { // negative axis range check
368 goto err;
369 }
370
371 const shape_elem_type positive_axis = _axis < 0 ? (_axis + input_shape_size_signed) : _axis;
372
373 if (!__allow_duplicate)
374 {
375 if (std::find(result.begin(), result.end(), positive_axis) != result.end())
376 { // find axis duplication
377 goto err;
378 }
379 }
380
381 result.push_back(positive_axis);
382 }
383
384out:
385 return result;
386
387err:
388 // TODO exception if wrong axis? need common function for throwing exceptions
389 throw std::range_error("DPNP Error: validate_axes() failed with axis check");
390}
