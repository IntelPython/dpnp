91template <typename _DataType>
92void get_shape_offsets_inkernel(const _DataType *shape,
93 size_t shape_size,
94 _DataType *offsets)
95{
96 size_t dim_prod_input = 1;
97 for (size_t i = 0; i < shape_size; ++i) {
98 long i_reverse = shape_size - 1 - i;
99 offsets[i_reverse] = dim_prod_input;
100 dim_prod_input *= shape[i_reverse];
101 }
102
103 return;
104}

120template <typename _DataType>
121void get_xyz_by_id(size_t idx,
122 size_t ndim,
123 const _DataType *offsets,
124 _DataType *xyz)
125{
126 size_t quotient;
127 size_t remainder = idx;
128
129 for (size_t i = 0; i < ndim; ++i) {
130 quotient = remainder / offsets[i];
131 remainder = remainder - quotient * offsets[i];
132 xyz[i] = quotient;
133 }
134 return;
135}

153template <typename _DataType>
154_DataType get_xyz_id_by_id_inkernel(size_t global_id,
155 const _DataType *offsets,
156 size_t offsets_size,
157 size_t axis)
158{
159 /* avoid warning unused variable*/
160 (void)offsets_size;
161
162 assert(axis < offsets_size);
163
164 _DataType xyz_id = 0;
165 long reminder = global_id;
166 for (size_t i = 0; i < axis + 1; ++i) {
167 const _DataType axis_val = offsets[i];
168 xyz_id = reminder / axis_val;
169 reminder = reminder % axis_val;
170 }
171
172 return xyz_id;
173}

188template <typename _DataType>
189size_t get_id_by_xyz_inkernel(const _DataType *xyz,
190 size_t xyz_size,
191 const _DataType *offsets)
192{
193 size_t global_id = 0;
194
195 /* calculate linear index based on reconstructed [x][y][z] */
196 for (size_t it = 0; it < xyz_size; ++it) {
197 global_id += (xyz[it] * offsets[it]);
198 }
199
200 return global_id;
201}

213static inline bool
214 broadcastable(const std::vector<shape_elem_type> &input_shape,
215 const std::vector<shape_elem_type> &output_shape)
216{
217 if (input_shape.size() > output_shape.size()) {
218 return false;
219 }
220
221 std::vector<shape_elem_type>::const_reverse_iterator irit =
222 input_shape.rbegin();
223 std::vector<shape_elem_type>::const_reverse_iterator orit =
224 output_shape.rbegin();
225 for (; irit != input_shape.rend(); ++irit, ++orit) {
226 if (*irit != 1 && *irit != *orit) {
227 return false;
228 }
229 }
230
231 return true;
232}

255template <typename _DataType>
256static inline bool array_equal(const _DataType *input1,
257 const size_t input1_size,
258 const _DataType *input2,
259 const size_t input2_size)
260{
261 if (input1_size != input2_size)
262 return false;
263

268 std::begin(input2_vec));

314template <typename _DataType>
315static inline std::vector<_DataType>
316 get_result_shape(const _DataType *input1_shape,
317 const size_t input1_shape_size,
318 const _DataType *input2_shape,
319 const size_t input2_shape_size)
320{
321 const size_t result_shape_size = (input2_shape_size > input1_shape_size)
322 ? input2_shape_size
323 : input1_shape_size;
324 std::vector<_DataType> result_shape;
325 result_shape.reserve(result_shape_size);
326
327 for (int irit1 = input1_shape_size - 1, irit2 = input2_shape_size - 1;
328 irit1 >= 0 || irit2 >= 0; --irit1, --irit2)
329 {
330 _DataType input1_val = (irit1 >= 0) ? input1_shape[irit1] : 1;
331 _DataType input2_val = (irit2 >= 0) ? input2_shape[irit2] : 1;
332
333 if (input1_val == input2_val || input1_val == 1) {
334 result_shape.insert(result_shape.begin(), input2_val);
335 }
336 else if (input2_val == 1) {
337 result_shape.insert(result_shape.begin(), input1_val);
338 }
339 else {
340 throw std::domain_error("DPNP Error: get_common_shape() failed "
341 "with input shapes check");
342 }
343 }
344
345 return result_shape;
346}

367static inline std::vector<shape_elem_type>
368 get_validated_axes(const std::vector<shape_elem_type> &__axes,
369 const size_t __shape_size,
370 const bool __allow_duplicate = false)
371{
372 std::vector<shape_elem_type> result;
373
374 if (__axes.empty()) {
375 goto out;
376 }
377
378 if (__axes.size() > __shape_size) {
379 goto err;
380 }
381
382 result.reserve(__axes.size());
383 for (std::vector<shape_elem_type>::const_iterator it = __axes.cbegin();
384 it != __axes.cend(); ++it)
385 {
386 const shape_elem_type _axis = *it;
387 const shape_elem_type input_shape_size_signed =
388 static_cast<shape_elem_type>(__shape_size);
389 if (_axis >= input_shape_size_signed) { // positive axis range check
390 goto err;
391 }
392
393 if (_axis < -input_shape_size_signed) { // negative axis range check
394 goto err;
395 }
396
397 const shape_elem_type positive_axis =
398 _axis < 0 ? (_axis + input_shape_size_signed) : _axis;
399
400 if (!__allow_duplicate) {
401 if (std::find(result.begin(), result.end(), positive_axis) !=
402 result.end()) { // find axis duplication
403 goto err;
404 }
405 }
406
407 result.push_back(positive_axis);
408 }
409
410out:
411 return result;
412
413err:
414 // TODO exception if wrong axis? need common function for throwing
415 // exceptions
416 throw std::range_error(
417 "DPNP Error: validate_axes() failed with axis check");
418}

430template <typename T>
431static inline void validate_type_for_device(const sycl::device &d)
432{
433 if constexpr (std::is_same_v<T, double>) {
434 if (!d.has(sycl::aspect::fp64)) {
435 throw std::runtime_error("Device " +
436 d.get_info<sycl::info::device::name>() +
437 " does not support type 'double'");
438 }
439 }
440 else if constexpr (std::is_same_v<T, std::complex<double>>) {
441 if (!d.has(sycl::aspect::fp64)) {
442 throw std::runtime_error(
443 "Device " + d.get_info<sycl::info::device::name>() +
444 " does not support type 'complex<double>'");
445 }
446 }
447 else if constexpr (std::is_same_v<T, sycl::half>) {
448 if (!d.has(sycl::aspect::fp16)) {
449 throw std::runtime_error("Device " +
450 d.get_info<sycl::info::device::name>() +
451 " does not support type 'half'");
452 }
453 }
454}
