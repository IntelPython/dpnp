57 template <typename _DataType>
58 void get_shape_offsets_inkernel(const _DataType* shape, size_t shape_size, _DataType* offsets)
59 {
60  size_t dim_prod_input = 1;
61  for (size_t i = 0; i < shape_size; ++i)
62  {
63  long i_reverse = shape_size - 1 - i;
64  offsets[i_reverse] = dim_prod_input;
65  dim_prod_input *= shape[i_reverse];
66  }
67 
68  return;
69 }

87 template <typename _DataType>
88 _DataType get_xyz_id_by_id_inkernel(size_t global_id, const _DataType* offsets, size_t offsets_size, size_t axis)
89 {
90  /* avoid warning unused variable*/
91  (void)offsets_size;
92 
93  assert(axis < offsets_size);
94 
95  _DataType xyz_id = 0;
96  long reminder = global_id;
97  for (size_t i = 0; i < axis + 1; ++i)
98  {
99  const _DataType axis_val = offsets[i];
100  xyz_id = reminder / axis_val;
101  reminder = reminder % axis_val;
102  }
103 
104  return xyz_id;
105 }

129  {

145 static inline bool broadcastable(const std::vector<size_t>& input_shape, const std::vector<size_t>& output_shape)
146 {
147  if (input_shape.size() > output_shape.size())
148  {
149  return false;
150  }
151 
152  std::vector<size_t>::const_reverse_iterator irit = input_shape.rbegin();
153  std::vector<size_t>::const_reverse_iterator orit = output_shape.rbegin();
154  for (; irit != input_shape.rend(); ++irit, ++orit)
155  {
156  if (*irit != 1 && *irit != *orit)
157  {
158  return false;
159  }
160  }
161 
162  return true;
163 }

189 static inline std::vector<size_t> get_result_shape(const size_t* input1_shape,
190  const size_t input1_shape_size,
191  const size_t* input2_shape,
192  const size_t input2_shape_size)
193 {
194  const size_t result_shape_size = (input2_shape_size > input1_shape_size) ? input2_shape_size : input1_shape_size;
195  std::vector<size_t> result_shape;
196  result_shape.reserve(result_shape_size);
197 
198  for (int irit1 = input1_shape_size - 1, irit2 = input2_shape_size - 1; irit1 >= 0 || irit2 >= 0; --irit1, --irit2)
199  {
200  size_t input1_val = (irit1 >= 0) ? input1_shape[irit1] : 1;
201  size_t input2_val = (irit2 >= 0) ? input2_shape[irit2] : 1;
202 
203  if (input1_val == input2_val || input1_val == 1)
204  {
205  result_shape.insert(result_shape.begin(), input2_val);
206  }
207  else if (input2_val == 1)
208  {
209  result_shape.insert(result_shape.begin(), input1_val);
210  }
211  else
212  {
213  throw std::domain_error("DPNP Error: get_common_shape() failed with input shapes check");
214  }
215  }
216 
217  return result_shape;
218 }

235 static inline std::vector<size_t>
236  get_validated_axes(const std::vector<long>& __axes, const size_t __shape_size, const bool __allow_duplicate = false)
237 {
238  std::vector<size_t> result;
239 
240  if (__axes.empty())
241  {
242  goto out;
243  }
244 
245  if (__axes.size() > __shape_size)
246  {
247  goto err;
248  }
249 
250  result.reserve(__axes.size());
251  for (std::vector<long>::const_iterator it = __axes.cbegin(); it != __axes.cend(); ++it)
252  {
253  const long _axis = *it;
254  const long input_shape_size_signed = static_cast<long>(__shape_size);
255  if (_axis >= input_shape_size_signed)
256  { // positive axis range check
257  goto err;
258  }
259 
260  if (_axis < -input_shape_size_signed)
261  { // negative axis range check
262  goto err;
263  }
264 
265  const size_t positive_axis = _axis < 0 ? (_axis + input_shape_size_signed) : _axis;
266 
267  if (!__allow_duplicate)
268  {
269  if (std::find(result.begin(), result.end(), positive_axis) != result.end())
270  { // find axis duplication
271  goto err;
272  }
273  }
274 
275  result.push_back(positive_axis);
276  }
277 
278 out:
279  return result;
280 
281 err:
282  // TODO exception if wrong axis? need common function for throwing exceptions
283  throw std::range_error("DPNP Error: validate_axes() failed with axis check");
284 }
