63template <typename _DataType>
64void get_shape_offsets_inkernel(const _DataType* shape, size_t shape_size, _DataType* offsets)
65{
66 size_t dim_prod_input = 1;
67 for (size_t i = 0; i < shape_size; ++i)
68 {
69 long i_reverse = shape_size - 1 - i;
70 offsets[i_reverse] = dim_prod_input;
71 dim_prod_input *= shape[i_reverse];
72 }
73
74 return;
75}

91template <typename _DataType>
92void get_xyz_by_id(size_t idx, size_t ndim, const _DataType* offsets, _DataType* xyz)
93{
94 size_t quotient;
95 size_t remainder = idx;
96
97 for (size_t i = 0; i < ndim; ++i)
98 {
99 quotient = remainder / offsets[i];
100 remainder = remainder - quotient * offsets[i];
101 xyz[i] = quotient;
102 }
103 return;
104}

122template <typename _DataType>
123_DataType get_xyz_id_by_id_inkernel(size_t global_id, const _DataType* offsets, size_t offsets_size, size_t axis)
124{
125 /* avoid warning unused variable*/
126 (void)offsets_size;
127
128 assert(axis < offsets_size);
129
130 _DataType xyz_id = 0;
131 long reminder = global_id;
132 for (size_t i = 0; i < axis + 1; ++i)
133 {
134 const _DataType axis_val = offsets[i];
135 xyz_id = reminder / axis_val;
136 reminder = reminder % axis_val;
137 }
138
139 return xyz_id;
140}

157template <typename _DataType>
158size_t get_id_by_xyz_inkernel(const _DataType* xyz, size_t xyz_size, const _DataType* offsets)
159{
160 size_t global_id = 0;
161
162 /* calculate linear index based on reconstructed [x][y][z] */
163 for (size_t it = 0; it < xyz_size; ++it)
164 {
165 global_id += (xyz[it] * offsets[it]);
166 }
167
168 return global_id;
169}

180static inline bool broadcastable(const std::vector<shape_elem_type>& input_shape,
181 const std::vector<shape_elem_type>& output_shape)
182{
183 if (input_shape.size() > output_shape.size())
184 {
185 return false;
186 }
187
188 std::vector<shape_elem_type>::const_reverse_iterator irit = input_shape.rbegin();
189 std::vector<shape_elem_type>::const_reverse_iterator orit = output_shape.rbegin();
190 for (; irit != input_shape.rend(); ++irit, ++orit)
191 {
192 if (*irit != 1 && *irit != *orit)
193 {
194 return false;
195 }
196 }
197
198 return true;
199}

276template <typename _DataType>
277static inline std::vector<_DataType> get_result_shape(const _DataType* input1_shape,
278 const size_t input1_shape_size,
279 const _DataType* input2_shape,
280 const size_t input2_shape_size)
281{
282 const size_t result_shape_size = (input2_shape_size > input1_shape_size) ? input2_shape_size : input1_shape_size;
283 std::vector<_DataType> result_shape;
284 result_shape.reserve(result_shape_size);
285
286 for (int irit1 = input1_shape_size - 1, irit2 = input2_shape_size - 1; irit1 >= 0 || irit2 >= 0; --irit1, --irit2)
287 {
288 _DataType input1_val = (irit1 >= 0) ? input1_shape[irit1] : 1;
289 _DataType input2_val = (irit2 >= 0) ? input2_shape[irit2] : 1;
290
291 if (input1_val == input2_val || input1_val == 1)
292 {
293 result_shape.insert(result_shape.begin(), input2_val);
294 }
295 else if (input2_val == 1)
296 {
297 result_shape.insert(result_shape.begin(), input1_val);
298 }
299 else
300 {
301 throw std::domain_error("DPNP Error: get_common_shape() failed with input shapes check");
302 }
303 }
304
305 return result_shape;
306}

323static inline std::vector<shape_elem_type> get_validated_axes(const std::vector<shape_elem_type>& __axes,
324 const size_t __shape_size,
325 const bool __allow_duplicate = false)
326{
327 std::vector<shape_elem_type> result;
328
329 if (__axes.empty())
330 {
331 goto out;
332 }
333
334 if (__axes.size() > __shape_size)
335 {
336 goto err;
337 }
338
339 result.reserve(__axes.size());
340 for (std::vector<shape_elem_type>::const_iterator it = __axes.cbegin(); it != __axes.cend(); ++it)
341 {
342 const shape_elem_type _axis = *it;
343 const shape_elem_type input_shape_size_signed = static_cast<shape_elem_type>(__shape_size);
344 if (_axis >= input_shape_size_signed)
345 { // positive axis range check
346 goto err;
347 }
348
349 if (_axis < -input_shape_size_signed)
350 { // negative axis range check
351 goto err;
352 }
353
354 const shape_elem_type positive_axis = _axis < 0 ? (_axis + input_shape_size_signed) : _axis;
355
356 if (!__allow_duplicate)
357 {
358 if (std::find(result.begin(), result.end(), positive_axis) != result.end())
359 { // find axis duplication
360 goto err;
361 }
362 }
363
364 result.push_back(positive_axis);
365 }
366
367out:
368 return result;
369
370err:
371 // TODO exception if wrong axis? need common function for throwing exceptions
372 throw std::range_error("DPNP Error: validate_axes() failed with axis check");
373}
