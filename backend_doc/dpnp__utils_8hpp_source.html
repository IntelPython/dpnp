56 template <typename _DataType>
57 void get_shape_offsets_inkernel(const _DataType* shape, size_t shape_size, _DataType* offsets)
58 {
59  size_t dim_prod_input = 1;
60  for (size_t i = 0; i < shape_size; ++i)
61  {
62  long i_reverse = shape_size - 1 - i;
63  offsets[i_reverse] = dim_prod_input;
64  dim_prod_input *= shape[i_reverse];
65  }
66 
67  return;
68 }

85 template <typename _DataType>
86 void get_xyz_by_id_inkernel(size_t global_id, const _DataType* offsets, size_t offsets_size, _DataType* xyz)
87 {
88  long reminder = global_id;
89  for (size_t axis = 0; axis < offsets_size; ++axis)
90  {
91  /* reconstruct [x][y][z] from given linear idx */
92  const _DataType axis_val = offsets[axis];
93  _DataType xyz_id = reminder / axis_val;
94  reminder = reminder % axis_val;
95  xyz[axis] = xyz_id;
96  }
97 
98  return;
99 }

145 static inline std::vector<size_t>
146  get_validated_axes(const std::vector<long>& __axes, const size_t __shape_size, const bool __allow_duplicate = false)
147 {
148  std::vector<size_t> result;
149 
150  if (__axes.empty())
151  {
152  goto out;
153  }
154 
155  if (__axes.size() > __shape_size)
156  {
157  goto err;
158  }
159 
160  result.reserve(__axes.size());
161  for (std::vector<long>::const_iterator it = __axes.cbegin(); it != __axes.cend(); ++it)
162  {
163  const long _axis = *it;
164  const long input_shape_size_signed = static_cast<long>(__shape_size);
165  if (_axis >= input_shape_size_signed)
166  { // positive axis range check
167  goto err;
168  }
169 
170  if (_axis < -input_shape_size_signed)
171  { // negative axis range check
172  goto err;
173  }
174 
175  const size_t positive_axis = _axis < 0 ? (_axis + input_shape_size_signed) : _axis;
176 
177  if (!__allow_duplicate)
178  {
179  if (std::find(result.begin(), result.end(), positive_axis) != result.end())
180  { // find axis duplication
181  goto err;
182  }
183  }
184 
185  result.push_back(positive_axis);
186  }
187 
188 out:
189  return result;
190 
191 err:
192  // TODO exception if wrong axis? need common function for throwing exceptions
193  throw std::range_error("DPNP Error: validate_axes() failed with axis check");
194 }
