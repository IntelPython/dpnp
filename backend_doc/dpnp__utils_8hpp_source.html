72template <typename _DataType>
73void get_shape_offsets_inkernel(const _DataType* shape, size_t shape_size, _DataType* offsets)
74{
75 size_t dim_prod_input = 1;
76 for (size_t i = 0; i < shape_size; ++i)
77 {
78 long i_reverse = shape_size - 1 - i;
79 offsets[i_reverse] = dim_prod_input;
80 dim_prod_input *= shape[i_reverse];
81 }
82
83 return;
84}

100template <typename _DataType>
101void get_xyz_by_id(size_t idx, size_t ndim, const _DataType* offsets, _DataType* xyz)
102{
103 size_t quotient;
104 size_t remainder = idx;
105
106 for (size_t i = 0; i < ndim; ++i)
107 {
108 quotient = remainder / offsets[i];
109 remainder = remainder - quotient * offsets[i];
110 xyz[i] = quotient;
111 }
112 return;
113}

131template <typename _DataType>
132_DataType get_xyz_id_by_id_inkernel(size_t global_id, const _DataType* offsets, size_t offsets_size, size_t axis)
133{
134 /* avoid warning unused variable*/
135 (void)offsets_size;
136
137 assert(axis < offsets_size);
138
139 _DataType xyz_id = 0;
140 long reminder = global_id;
141 for (size_t i = 0; i < axis + 1; ++i)
142 {
143 const _DataType axis_val = offsets[i];
144 xyz_id = reminder / axis_val;
145 reminder = reminder % axis_val;
146 }
147
148 return xyz_id;
149}

166template <typename _DataType>
167size_t get_id_by_xyz_inkernel(const _DataType* xyz, size_t xyz_size, const _DataType* offsets)
168{
169 size_t global_id = 0;
170
171 /* calculate linear index based on reconstructed [x][y][z] */
172 for (size_t it = 0; it < xyz_size; ++it)
173 {
174 global_id += (xyz[it] * offsets[it]);
175 }
176
177 return global_id;
178}

189static inline bool broadcastable(const std::vector<shape_elem_type>& input_shape,
190 const std::vector<shape_elem_type>& output_shape)
191{
192 if (input_shape.size() > output_shape.size())
193 {
194 return false;
195 }
196
197 std::vector<shape_elem_type>::const_reverse_iterator irit = input_shape.rbegin();
198 std::vector<shape_elem_type>::const_reverse_iterator orit = output_shape.rbegin();
199 for (; irit != input_shape.rend(); ++irit, ++orit)
200 {
201 if (*irit != 1 && *irit != *orit)
202 {
203 return false;
204 }
205 }
206
207 return true;
208}

286template <typename _DataType>
287static inline std::vector<_DataType> get_result_shape(const _DataType* input1_shape,
288 const size_t input1_shape_size,
289 const _DataType* input2_shape,
290 const size_t input2_shape_size)
291{
292 const size_t result_shape_size = (input2_shape_size > input1_shape_size) ? input2_shape_size : input1_shape_size;
293 std::vector<_DataType> result_shape;
294 result_shape.reserve(result_shape_size);
295
296 for (int irit1 = input1_shape_size - 1, irit2 = input2_shape_size - 1; irit1 >= 0 || irit2 >= 0; --irit1, --irit2)
297 {
298 _DataType input1_val = (irit1 >= 0) ? input1_shape[irit1] : 1;
299 _DataType input2_val = (irit2 >= 0) ? input2_shape[irit2] : 1;
300
301 if (input1_val == input2_val || input1_val == 1)
302 {
303 result_shape.insert(result_shape.begin(), input2_val);
304 }
305 else if (input2_val == 1)
306 {
307 result_shape.insert(result_shape.begin(), input1_val);
308 }
309 else
310 {
311 throw std::domain_error("DPNP Error: get_common_shape() failed with input shapes check");
312 }
313 }
314
315 return result_shape;
316}

333static inline std::vector<shape_elem_type> get_validated_axes(const std::vector<shape_elem_type>& __axes,
334 const size_t __shape_size,
335 const bool __allow_duplicate = false)
336{
337 std::vector<shape_elem_type> result;
338
339 if (__axes.empty())
340 {
341 goto out;
342 }
343
344 if (__axes.size() > __shape_size)
345 {
346 goto err;
347 }
348
349 result.reserve(__axes.size());
350 for (std::vector<shape_elem_type>::const_iterator it = __axes.cbegin(); it != __axes.cend(); ++it)
351 {
352 const shape_elem_type _axis = *it;
353 const shape_elem_type input_shape_size_signed = static_cast<shape_elem_type>(__shape_size);
354 if (_axis >= input_shape_size_signed)
355 { // positive axis range check
356 goto err;
357 }
358
359 if (_axis < -input_shape_size_signed)
360 { // negative axis range check
361 goto err;
362 }
363
364 const shape_elem_type positive_axis = _axis < 0 ? (_axis + input_shape_size_signed) : _axis;
365
366 if (!__allow_duplicate)
367 {
368 if (std::find(result.begin(), result.end(), positive_axis) != result.end())
369 { // find axis duplication
370 goto err;
371 }
372 }
373
374 result.push_back(positive_axis);
375 }
376
377out:
378 return result;
379
380err:
381 // TODO exception if wrong axis? need common function for throwing exceptions
382 throw std::range_error("DPNP Error: validate_axes() failed with axis check");
383}
