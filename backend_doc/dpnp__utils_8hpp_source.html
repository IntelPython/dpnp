87template <typename _DataType>
88void get_shape_offsets_inkernel(const _DataType* shape, size_t shape_size, _DataType* offsets)
89{
90 size_t dim_prod_input = 1;
91 for (size_t i = 0; i < shape_size; ++i)
92 {
93 long i_reverse = shape_size - 1 - i;
94 offsets[i_reverse] = dim_prod_input;
95 dim_prod_input *= shape[i_reverse];
96 }
97
98 return;
99}

115template <typename _DataType>
116void get_xyz_by_id(size_t idx, size_t ndim, const _DataType* offsets, _DataType* xyz)
117{
118 size_t quotient;
119 size_t remainder = idx;
120
121 for (size_t i = 0; i < ndim; ++i)
122 {
123 quotient = remainder / offsets[i];
124 remainder = remainder - quotient * offsets[i];
125 xyz[i] = quotient;
126 }
127 return;
128}

146template <typename _DataType>
147_DataType get_xyz_id_by_id_inkernel(size_t global_id, const _DataType* offsets, size_t offsets_size, size_t axis)
148{
149 /* avoid warning unused variable*/
150 (void)offsets_size;
151
152 assert(axis < offsets_size);
153
154 _DataType xyz_id = 0;
155 long reminder = global_id;
156 for (size_t i = 0; i < axis + 1; ++i)
157 {
158 const _DataType axis_val = offsets[i];
159 xyz_id = reminder / axis_val;
160 reminder = reminder % axis_val;
161 }
162
163 return xyz_id;
164}

181template <typename _DataType>
182size_t get_id_by_xyz_inkernel(const _DataType* xyz, size_t xyz_size, const _DataType* offsets)
183{
184 size_t global_id = 0;
185
186 /* calculate linear index based on reconstructed [x][y][z] */
187 for (size_t it = 0; it < xyz_size; ++it)
188 {
189 global_id += (xyz[it] * offsets[it]);
190 }
191
192 return global_id;
193}

204static inline bool broadcastable(const std::vector<shape_elem_type>& input_shape,
205 const std::vector<shape_elem_type>& output_shape)
206{
207 if (input_shape.size() > output_shape.size())
208 {
209 return false;
210 }
211
212 std::vector<shape_elem_type>::const_reverse_iterator irit = input_shape.rbegin();
213 std::vector<shape_elem_type>::const_reverse_iterator orit = output_shape.rbegin();
214 for (; irit != input_shape.rend(); ++irit, ++orit)
215 {
216 if (*irit != 1 && *irit != *orit)
217 {
218 return false;
219 }
220 }
221
222 return true;
223}

301template <typename _DataType>
302static inline std::vector<_DataType> get_result_shape(const _DataType* input1_shape,
303 const size_t input1_shape_size,
304 const _DataType* input2_shape,
305 const size_t input2_shape_size)
306{
307 const size_t result_shape_size = (input2_shape_size > input1_shape_size) ? input2_shape_size : input1_shape_size;
308 std::vector<_DataType> result_shape;
309 result_shape.reserve(result_shape_size);
310
311 for (int irit1 = input1_shape_size - 1, irit2 = input2_shape_size - 1; irit1 >= 0 || irit2 >= 0; --irit1, --irit2)
312 {
313 _DataType input1_val = (irit1 >= 0) ? input1_shape[irit1] : 1;
314 _DataType input2_val = (irit2 >= 0) ? input2_shape[irit2] : 1;
315
316 if (input1_val == input2_val || input1_val == 1)
317 {
318 result_shape.insert(result_shape.begin(), input2_val);
319 }
320 else if (input2_val == 1)
321 {
322 result_shape.insert(result_shape.begin(), input1_val);
323 }
324 else
325 {
326 throw std::domain_error("DPNP Error: get_common_shape() failed with input shapes check");
327 }
328 }
329
330 return result_shape;
331}

348static inline std::vector<shape_elem_type> get_validated_axes(const std::vector<shape_elem_type>& __axes,
349 const size_t __shape_size,
350 const bool __allow_duplicate = false)
351{
352 std::vector<shape_elem_type> result;
353
354 if (__axes.empty())
355 {
356 goto out;
357 }
358
359 if (__axes.size() > __shape_size)
360 {
361 goto err;
362 }
363
364 result.reserve(__axes.size());
365 for (std::vector<shape_elem_type>::const_iterator it = __axes.cbegin(); it != __axes.cend(); ++it)
366 {
367 const shape_elem_type _axis = *it;
368 const shape_elem_type input_shape_size_signed = static_cast<shape_elem_type>(__shape_size);
369 if (_axis >= input_shape_size_signed)
370 { // positive axis range check
371 goto err;
372 }
373
374 if (_axis < -input_shape_size_signed)
375 { // negative axis range check
376 goto err;
377 }
378
379 const shape_elem_type positive_axis = _axis < 0 ? (_axis + input_shape_size_signed) : _axis;
380
381 if (!__allow_duplicate)
382 {
383 if (std::find(result.begin(), result.end(), positive_axis) != result.end())
384 { // find axis duplication
385 goto err;
386 }
387 }
388
389 result.push_back(positive_axis);
390 }
391
392out:
393 return result;
394
395err:
396 // TODO exception if wrong axis? need common function for throwing exceptions
397 throw std::range_error("DPNP Error: validate_axes() failed with axis check");
398}

410template <typename T>
411static inline void validate_type_for_device(const sycl::device &d)
412{
413 if constexpr (std::is_same_v<T, double>) {
414 if (!d.has(sycl::aspect::fp64)) {
415 throw std::runtime_error("Device " +
416 d.get_info<sycl::info::device::name>() +
417 " does not support type 'double'");
418 }
419 }
420 else if constexpr (std::is_same_v<T, std::complex<double>>) {
421 if (!d.has(sycl::aspect::fp64)) {
422 throw std::runtime_error("Device " +
423 d.get_info<sycl::info::device::name>() +
424 " does not support type 'complex<double>'");
425 }
426 }
427 else if constexpr (std::is_same_v<T, sycl::half>) {
428 if (!d.has(sycl::aspect::fp16)) {
429 throw std::runtime_error("Device " +
430 d.get_info<sycl::info::device::name>() +
431 " does not support type 'half'");
432 }
433 }
434}
