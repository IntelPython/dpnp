62 void get_shape_offsets_inkernel(const _DataType* shape, size_t shape_size, _DataType* offsets)
63 {
64  size_t dim_prod_input = 1;

67  long i_reverse = shape_size - 1 - i;

120 template <typename _DataType>
121 _DataType get_xyz_id_by_id_inkernel(size_t global_id, const _DataType* offsets, size_t offsets_size, size_t axis)
122 {
123  /* avoid warning unused variable*/
124  (void)offsets_size;
125 
126  assert(axis < offsets_size);
127 
128  _DataType xyz_id = 0;
129  long reminder = global_id;
130  for (size_t i = 0; i < axis + 1; ++i)
131  {
132  const _DataType axis_val = offsets[i];
133  xyz_id = reminder / axis_val;
134  reminder = reminder % axis_val;
135  }
136 
137  return xyz_id;

178 static inline bool broadcastable(const std::vector<size_t>& input_shape, const std::vector<size_t>& output_shape)
179 {
180  if (input_shape.size() > output_shape.size())
181  {
182  return false;
183  }
184 
185  std::vector<size_t>::const_reverse_iterator irit = input_shape.rbegin();
186  std::vector<size_t>::const_reverse_iterator orit = output_shape.rbegin();
187  for (; irit != input_shape.rend(); ++irit, ++orit)
188  {
189  if (*irit != 1 && *irit != *orit)
190  {
191  return false;
192  }
193  }
194 
195  return true;
196 }

222 static inline std::vector<size_t> get_result_shape(const size_t* input1_shape,
223  const size_t input1_shape_size,
224  const size_t* input2_shape,
225  const size_t input2_shape_size)
226 {
227  const size_t result_shape_size = (input2_shape_size > input1_shape_size) ? input2_shape_size : input1_shape_size;
228  std::vector<size_t> result_shape;
229  result_shape.reserve(result_shape_size);
230 
231  for (int irit1 = input1_shape_size - 1, irit2 = input2_shape_size - 1; irit1 >= 0 || irit2 >= 0; --irit1, --irit2)
232  {
233  size_t input1_val = (irit1 >= 0) ? input1_shape[irit1] : 1;
234  size_t input2_val = (irit2 >= 0) ? input2_shape[irit2] : 1;
235 
236  if (input1_val == input2_val || input1_val == 1)
237  {
238  result_shape.insert(result_shape.begin(), input2_val);
239  }
240  else if (input2_val == 1)
241  {
242  result_shape.insert(result_shape.begin(), input1_val);
243  }
244  else
245  {
246  throw std::domain_error("DPNP Error: get_common_shape() failed with input shapes check");
247  }
248  }
249 
250  return result_shape;
251 }

268 static inline std::vector<size_t>
269  get_validated_axes(const std::vector<long>& __axes, const size_t __shape_size, const bool __allow_duplicate = false)
270 {
271  std::vector<size_t> result;
272 
273  if (__axes.empty())
274  {
275  goto out;
276  }
277 
278  if (__axes.size() > __shape_size)
279  {
280  goto err;
281  }
282 
283  result.reserve(__axes.size());
284  for (std::vector<long>::const_iterator it = __axes.cbegin(); it != __axes.cend(); ++it)
285  {
286  const long _axis = *it;
287  const long input_shape_size_signed = static_cast<long>(__shape_size);
288  if (_axis >= input_shape_size_signed)
289  { // positive axis range check
290  goto err;
291  }
292 
293  if (_axis < -input_shape_size_signed)
294  { // negative axis range check
295  goto err;
296  }
297 
298  const size_t positive_axis = _axis < 0 ? (_axis + input_shape_size_signed) : _axis;
299 
300  if (!__allow_duplicate)
301  {
302  if (std::find(result.begin(), result.end(), positive_axis) != result.end())
303  { // find axis duplication
304  goto err;
305  }
306  }
307 
308  result.push_back(positive_axis);
309  }
310 
311 out:
312  return result;
313 
314 err:
315  // TODO exception if wrong axis? need common function for throwing exceptions
316  throw std::range_error("DPNP Error: validate_axes() failed with axis check");
317 }
