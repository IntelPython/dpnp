56 template <typename _DataType>
57 void get_shape_offsets_inkernel(const _DataType* shape, size_t shape_size, _DataType* offsets)
58 {
59  size_t dim_prod_input = 1;
60  for (size_t i = 0; i < shape_size; ++i)
61  {
62  long i_reverse = shape_size - 1 - i;
63  offsets[i_reverse] = dim_prod_input;
64  dim_prod_input *= shape[i_reverse];
65  }
66 
67  return;
68 }

85 template <typename _DataType>
86 void get_xyz_by_id_inkernel(size_t global_id, const _DataType* offsets, size_t offsets_size, _DataType* xyz)
87 {
88  long reminder = global_id;
89  for (size_t axis = 0; axis < offsets_size; ++axis)
90  {
91  /* reconstruct [x][y][z] from given linear idx */
92  const _DataType axis_val = offsets[axis];
93  _DataType xyz_id = reminder / axis_val;
94  reminder = reminder % axis_val;
95  xyz[axis] = xyz_id;
96  }
97 
98  return;
99 }

118 {

121  /* calculate linear index based on reconstructed [x][y][z] */

123  {
