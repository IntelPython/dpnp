43template <typename _DataType>
44class DPNPC_ptr_adapter final
45{
46 DPCTLSyclQueueRef queue_ref;
47 sycl::queue queue;
48 void* aux_ptr = nullptr;
49 void* orig_ptr = nullptr;
50 size_t size_in_bytes = 0;
51 bool allocated = false;
52 bool target_no_queue = false;
53 bool copy_back = false;
54 const bool verbose = false;
55
56public:
57 DPNPC_ptr_adapter() = delete;
58
59 DPNPC_ptr_adapter(DPCTLSyclQueueRef q_ref,
60 const void* src_ptr,
61 const size_t size,
62 bool target_no_sycl = false,
63 bool copy_back_request = false)
64 {
65 queue_ref = q_ref;
66 queue = *(reinterpret_cast<sycl::queue*>(queue_ref));
67 target_no_queue = target_no_sycl;
68 copy_back = copy_back_request;
69 orig_ptr = const_cast<void*>(src_ptr);
70 size_in_bytes = size * sizeof(_DataType);
71
72 // enum class alloc { host = 0, device = 1, shared = 2, unknown = 3 };
73 sycl::usm::alloc src_ptr_type = sycl::usm::alloc::unknown;
74 src_ptr_type = sycl::get_pointer_type(src_ptr, queue.get_context());
75 if (verbose)
76 {
77 std::cerr << "DPNPC_ptr_converter:";
78 std::cerr << "\n\t target_no_queue=" << target_no_queue;
79 std::cerr << "\n\t copy_back=" << copy_back;
80 std::cerr << "\n\t pointer=" << src_ptr;
81 std::cerr << "\n\t size=" << size;
82 std::cerr << "\n\t size_in_bytes=" << size_in_bytes;
83 std::cerr << "\n\t pointer type=" << (long)src_ptr_type;
84 std::cerr << "\n\t queue inorder=" << queue.is_in_order();
85 std::cerr << "\n\t queue is_host=" << queue.is_host();
86 std::cerr << "\n\t queue device is_host=" << queue.get_device().is_host();
87 std::cerr << "\n\t queue device is_cpu=" << queue.get_device().is_cpu();
88 std::cerr << "\n\t queue device is_gpu=" << queue.get_device().is_gpu();
89 std::cerr << "\n\t queue device is_accelerator=" << queue.get_device().is_accelerator();
90 std::cerr << std::endl;
91 }
92
93 if (is_memcpy_required(src_ptr_type))
94 {
95 aux_ptr = dpnp_memory_alloc_c(queue_ref, size_in_bytes);
96 dpnp_memory_memcpy_c(queue_ref, aux_ptr, src_ptr, size_in_bytes);
97 allocated = true;
98 if (verbose)
99 {
100 std::cerr << "DPNPC_ptr_converter::alloc and copy memory"
101 << " from=" << src_ptr << " to=" << aux_ptr << " size_in_bytes=" << size_in_bytes
102 << std::endl;
103 }
104 }
105 else
106 {
107 aux_ptr = const_cast<void*>(src_ptr);
108 }
109 }
110
111 ~DPNPC_ptr_adapter()
112 {
113 if (allocated)
114 {
115 if (verbose)
116 {
117 std::cerr << "DPNPC_ptr_converter::free_memory at=" << aux_ptr << std::endl;
118 }
119
120 if (copy_back)
121 {
122 copy_data_back();
123 }
124
125 dpnp_memory_free_c(queue_ref, aux_ptr);
126 }
127 }
128
129 bool is_memcpy_required(sycl::usm::alloc src_ptr_type)
130 {
131 if (target_no_queue || queue.get_device().is_gpu())
132 {
133 if (src_ptr_type == sycl::usm::alloc::unknown)
134 {
135 return true;
136 }
137 else if (target_no_queue && src_ptr_type == sycl::usm::alloc::device)
138 {
139 return true;
140 }
141 }
142
143 return false;
144 }
145
146 _DataType* get_ptr() const
147 {
148 return reinterpret_cast<_DataType*>(aux_ptr);
149 }
150
151 void copy_data_back() const
152 {
153 if (verbose)
154 {
155 std::cerr << "DPNPC_ptr_converter::copy_data_back:"
156 << " from=" << aux_ptr << " to=" << orig_ptr << " size_in_bytes=" << size_in_bytes << std::endl;
157 }
158
159 dpnp_memory_memcpy_c(queue_ref, orig_ptr, aux_ptr, size_in_bytes);
160 }
161};
