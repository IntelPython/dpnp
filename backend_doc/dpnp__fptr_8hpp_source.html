107template <DPNPFuncType FT1, DPNPFuncType FT2>
108static constexpr DPNPFuncType populate_func_types()
109{
110 if constexpr (FT1 == DPNPFuncType::DPNP_FT_NONE)
111 {
112 throw std::runtime_error("Templated enum value of FT1 is None");
113 }
114 else if constexpr (FT2 == DPNPFuncType::DPNP_FT_NONE)
115 {
116 throw std::runtime_error("Templated enum value of FT2 is None");
117 }
118 return (FT1 < FT2) ? FT2 : FT1;
119}
