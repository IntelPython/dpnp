38 template <typename _DataType>
39 void custom_blas_gemm_c(void* array1_in, void* array2_in, void* result1, size_t size_m, size_t size_n, size_t size_k)
40 {
41  cl::sycl::event event;
42  _DataType* array_1 = reinterpret_cast<_DataType*>(array1_in);
43  _DataType* array_2 = reinterpret_cast<_DataType*>(array2_in);
44  _DataType* result = reinterpret_cast<_DataType*>(result1);
45 
46  // input1: M x K
47  // input2: K x N
48  // result: M x N
49  const size_t dim_m = size_m; // shape1.front(); // First dimensions of array1
50  const size_t dim_n = size_n; // shape2.back(); // Last dimensions of array2
51  const size_t dim_k = size_k; // shape1.back(); // First dimensions of array2
52 
53  cl::sycl::range<2> gws(dim_m, dim_n); // dimensions are: "i" and "j"
54  event = DPNP_QUEUE.submit([&](cl::sycl::handler& cgh) {
55  cgh.parallel_for<class custom_blas_gemm_c_kernel<_DataType> >(
56  gws,
57  [=](cl::sycl::id<2> global_id)
58  {
59  size_t i = global_id[0]; //for (size_t i = 0; i < size; ++i)
60  {
61  size_t j = global_id[1]; //for (size_t j = 0; j < size; ++j)
62  {
63  _DataType acc = _DataType(0);
64  for (size_t k = 0; k < dim_k; ++k)
65  {
66  const size_t index_1 = i * dim_k + k;
67  const size_t index_2 = k * dim_n + j;
68  acc += array_1[index_1] * array_2[index_2];
69  }
70  const size_t index_result = i * dim_n + j;
71  result[index_result] = acc;
72  }
73  }
74  }); // parallel_for
75  }); // queue.submit
76 
77  event.wait();
78 }
