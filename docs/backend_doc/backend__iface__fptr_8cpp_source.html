49 DPNPFuncData_t get_dpnp_function_ptr(DPNPFuncName func_name, DPNPFuncType first_type, DPNPFuncType second_type)
50 {
51  DPNPFuncType local_second_type = (second_type == DPNPFuncType::DPNP_FT_NONE) ? first_type : second_type;
52 
53  func_map_t::const_iterator func_it = func_map.find(func_name);
54  if (func_it == func_map.cend())
55  {
56  throw std::runtime_error("Intel NumPy Error: Unsupported function call."); // TODO print Function ID
57  }
58 
59  const map_1p_t& type1_map = func_it->second;
60  map_1p_t::const_iterator type1_map_it = type1_map.find(first_type);
61  if (type1_map_it == type1_map.cend())
62  {
63  throw std::runtime_error(
64  "Intel NumPy Error: Function ID with unsupported first parameter type."); // TODO print Function ID
65  }
66 
67  const map_2p_t& type2_map = type1_map_it->second;
68  map_2p_t::const_iterator type2_map_it = type2_map.find(local_second_type);
69  if (type2_map_it == type2_map.cend())
70  {
71  throw std::runtime_error(
72  "Intel NumPy Error: Function ID with unsupported second parameter type."); // TODO print Function ID
73  }
74 
75  DPNPFuncData_t func_info = type2_map_it->second;
76 
77  return func_info;
78 }

80 void* get_backend_function_name(const char* func_name, const char* type_name)
81 {
83  const char* supported_func_name = "dpnp_dot";
84  const char* supported_type1_name = "double";
85  const char* supported_type2_name = "float";
86  const char* supported_type3_name = "long";
87  const char* supported_type4_name = "int";
88 
90  if (!strncmp(func_name, supported_func_name, strlen(supported_func_name)))
91  {
92  if (!strncmp(type_name, supported_type1_name, strlen(supported_type1_name)))
93  {
94  return reinterpret_cast<void*>(mkl_blas_dot_c<double>);
95  }
96  else if (!strncmp(type_name, supported_type2_name, strlen(supported_type2_name)))
97  {
98  return reinterpret_cast<void*>(mkl_blas_dot_c<float>);
99  }
100  else if (!strncmp(type_name, supported_type3_name, strlen(supported_type3_name)))
101  {
102  return reinterpret_cast<void*>(custom_blas_dot_c<long>);
103  }
104  else if (!strncmp(type_name, supported_type4_name, strlen(supported_type4_name)))
105  {
106  return reinterpret_cast<void*>(custom_blas_dot_c<int>);
107  }
108  }
109 
110  throw std::runtime_error("Intel NumPy Error: Unsupported function call");
111 }

117 size_t operator-(DPNPFuncType lhs, DPNPFuncType rhs)
118 {
119  size_t lhs_base = static_cast<size_t>(lhs);
120  size_t rhs_base = static_cast<size_t>(rhs);
121 
122  size_t result = lhs_base - rhs_base;
123 
124  return result;
125 }
