41 {
42  _DataType* array_1 = reinterpret_cast<_DataType*>(array1_in);
43  _DataType* result = reinterpret_cast<_DataType*>(result1);
44 
45  if (!nrows || !ncols)
46  {

50  auto policy = oneapi::dpl::execution::make_device_policy<class custom_cov_c_kernel<_DataType>>(DPNP_QUEUE);
51 
52  _DataType* mean = reinterpret_cast<_DataType*>(dpnp_memory_alloc_c(nrows * sizeof(_DataType)));
53  for (size_t i = 0; i < nrows; ++i)
54  {
55  _DataType* row_start = array_1 + ncols * i;
56  mean[i] = std::reduce(policy, row_start, row_start + ncols, _DataType(0), std::plus<_DataType>()) / ncols;
57  }
58  policy.queue().wait();
59

70  for (size_t i = 0; i < nrows; ++i)
71  {
72  size_t offset = ncols * i;
73  _DataType* row_start = array_1 + offset;
74  std::transform(policy, row_start, row_start + ncols, temp + offset, [=](_DataType x) { return x - mean[i]; });
75  }

123  gws,

125  {
