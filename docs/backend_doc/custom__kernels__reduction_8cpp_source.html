38 void custom_sum_c(void* array1_in, void* result1, size_t size)
39 {
40  _DataType* array_1 = reinterpret_cast<_DataType*>(array1_in);
41  _DataType* result = reinterpret_cast<_DataType*>(result1);
42 
43 #if 1 // naive algorithm
44  // cl::sycl::range<1> gws(size);
45  auto policy = oneapi::dpl::execution::make_device_policy<custom_sum_c_kernel<_DataType>>(DPNP_QUEUE);
46 
47  // sycl::buffer<_DataType, 1> array_1_buf(array_1, gws);
48  // auto it_begin = oneapi::dpl::begin(array_1_buf);
49  // auto it_end = oneapi::dpl::end(array_1_buf);
50 
51  _DataType accumulator = 0;
52  accumulator = std::reduce(policy, array_1, array_1 + size, _DataType(0), std::plus<_DataType>());
53 
54  policy.queue().wait();
55 
56 #if 0 // verification
57  accumulator = 0;
58  for (size_t i = 0; i < size; ++i)
59  {
60  accumulator += array_1[i];
61  }
62  // std::cout << "result: " << accumulator << std::endl;
63 #endif
64 
65  result[0] = accumulator;
66 
67  return;
68 
69 #else // naive algorithm
70 
71  cl::sycl::event event;
72 
73  cl::sycl::buffer<_DataType, 1> work_buffer(array_1, cl::sycl::range<1>(size));
74  work_buffer.set_final_data(nullptr);
75 
76  auto group_max_size = DPNP_QUEUE.get_device().get_info<cl::sycl::info::device::max_work_group_size>();
77  size_t local = std::min(size, group_max_size);
78  size_t length = size;
79 
84  do
85  {
86  std::cout << "====================================\n"
87  << "local=" << local << "\n"
88  << "length=" << length << "\n"
89  << std::flush;
90  auto kernel_func = [length, local, &work_buffer](cl::sycl::handler& cgh) {
91  /* Two accessors are used: one to the buffer that is being reduced,
92  * and a second to local memory, used to store intermediate data. */
93  auto work_buffer_ptr = work_buffer.template get_access<cl::sycl::access::mode::read_write>(cgh);
94  // local, read/write memory to make barriers work
95  cl::sycl::accessor<_DataType, 1, cl::sycl::access::mode::read_write, cl::sycl::access::target::local>
96  local_mem_buf(cl::sycl::range<1>(local), cgh);
97 
98  auto kernel_parallel_for_func = [work_buffer_ptr, local_mem_buf, local, length](cl::sycl::nd_item<1> id) {
99  size_t globalid = id.get_global_id(0);
100  size_t localid = id.get_local_id(0);
101 
102  /* All threads collectively read from global memory into local.
103  * The barrier ensures all threads' IO is resolved before
104  * execution continues (strictly speaking, all threads within
105  * a single work-group - there is no co-ordination between
106  * work-groups, only work-items). */
107  if (globalid < length)
108  {
109  local_mem_buf[localid] = work_buffer_ptr[globalid];
110  }
111  id.barrier(cl::sycl::access::fence_space::local_space);
112 
113  /* Apply the reduction operation between the current local
114  * id and the one on the other half of the vector. */
115  if (globalid < length)
116  {
117  int min = (length < local) ? length : local;
118  for (size_t offset = min / 2; offset > 0; offset /= 2)
119  {
120  if (localid < offset)
121  {
122  local_mem_buf[localid] += local_mem_buf[localid + offset];
123  }
124  id.barrier(cl::sycl::access::fence_space::local_space);
125  }
126  /* The final result will be stored in local id 0. */
127  if (localid == 0)
128  {
129  work_buffer_ptr[id.get_group(0)] = local_mem_buf[localid];
130  }
131  }
132  };
133 
138  cl::sycl::nd_range<1> work_range{cl::sycl::range<1>{std::max(length, local)},
139  cl::sycl::range<1>{std::min(length, local)}};
140 
141  cgh.parallel_for<custom_sum_c_kernel<_DataType>>(work_range, kernel_parallel_for_func);
142  };
143 
144  DPNP_QUEUE.submit(kernel_func);
145  /* At this point, you could queue::wait_and_throw() to ensure that
146  * errors are caught quickly. However, this would likely impact
147  * performance negatively. */
148  length = length / local;
149  } while (length > 1);
150 
151  /* It is always sensible to wrap host accessors in their own scope as
152  * kernels using the buffers they access are blocked for the length
153  * of the accessor's lifetime. */
154  auto work_buffer_ro = work_buffer.template get_access<cl::sycl::access::mode::read>();
155  result[0] = work_buffer_ro[0];
156 
157  DPNP_QUEUE.wait();
158 
159 #endif // naive algorithm
160 }
